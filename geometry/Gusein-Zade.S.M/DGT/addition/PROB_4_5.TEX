\begin{problem_*}
Доказать, что $T\otimes S$ действительно является тензором.
\end{problem_*}

\begin{problem_*}
Пусть $\delta_{ij}=\left\{
\begin{array}{cc}
1 & \mbox{ при }i=j,\\
0 & \mbox{ при }i\ne j.
\end{array}
\right.$
Доказать, что $\delta_{ij}$ не является тензором (типа $(0, 2)$).
Доказать, что $\delta_{ij}$ не является тензором никакого типа.
\end{problem_*}

\begin{problem_*}
Пусть $\delta^i_{j}=\left\{
\begin{array}{cc}
1 & \mbox{ при }i=j,\\
0 & \mbox{ при }i\ne j.
\end{array}
\right.$
Доказать, что $\delta^i_{j}$ является тензором типа $(1, 1)$.
Что за оператор (в касательном пространстве к рассматриваемому
многообразию) отвечает этому тензору?
\end{problem_*}

\begin{problem_*}
Пусть $f$~--- функция на многообразии $M^n$. Доказать, что
$v^i={{\partial f}\over{\partial x^i}}$ вектором, вообще говоря,
не является.
\end{problem_*}

\begin{problem_*}
Пусть $u^i$ и $v^j$~--- вектора (в одной точке). Является ли вектором
$w^i = u^i v^i$?
\end{problem_*}

\begin{problem_*}
Пусть $u^i$ ($i=1, 2$)~--- вектор в некоторой точке поверхности.
Является ли вектором $v^i$, где $v^1=-u^2$, $v^2=u^1$?
\end{problem_*}

\begin{problem_*}
Пусть $f$~--- функция на многообразии $M^n$. Доказать, что
${{\partial^2 f}\over{\partial x^i \partial x^j}}$ тензором
(типа $(0, 2)$), вообще говоря, не является.
\end{problem_*}

\begin{problem_*}
Пусть $f$~--- функция на многообразии $M^n$. Доказать,
что в критических точках функции $f$ (т.е. в точках, в которых
${{\partial f}\over{\partial x^i}}=0$)
${{\partial^2 f}\over{\partial x^i \partial x^j}}$ является тензором
типа $(0, 2)$.
\end{problem_*}

\begin{problem_*}
Пусть $v^i=v^i(x)$~--- векторное поле на многообразии $M^n$. Доказать, что
${{\partial v^i}\over{\partial x^j}}$ тензором (типа $(1, 1)$), вообще
говоря, не является.
\end{problem_*}

\begin{problem_*}
Пусть $v^i=v^i(x)$~--- векторное поле на многообразии $M^n$. Доказать, что
$T^{ij}={{\partial v^i}\over{\partial x^j}}$ тензором (типа $(2, 0)$), вообще
говоря, не является.
\end{problem_*}

\begin{problem}
Пусть $u^i=u^i(x)$ и $v^i=v^i(x)$~--- векторные поля на многообразии $M^n$.
Является ли векторным полем $w^{i} = u^j{{\partial v^i}\over{\partial x^j}}-
v^j{{\partial u^i}\over{\partial x^j}}$.
\end{problem}

\begin{problem_*}
Пусть $\alpha_i=\alpha_i(x)$~--- ковекторное поле на многообразии $M^n$.
Доказать, что ${{\partial \alpha_i}\over{\partial x^j}}$ тензором (типа
$(0, 2)$), вообще говоря, не является.
\end{problem_*}

\begin{problem_*}
Пусть $T^{ij}_{kl}$~--- тензор типа $(2, 2)$. Доказать, что
${\widehat T}^i_\ell=T^{ij}_{j\ell}$ является тензором (типа
$(1, 1)$).
\end{problem_*}

\begin{problem}
Пусть $v^i=v^i(x)$~--- векторное поле на многообразии $M^n$. Доказать, что
в точках, в которых $v_i$ обращается в ноль, ${{\partial v^i}\over
{\partial x^j}}$ является тензором типа $(1, 1)$.
\end{problem}

\begin{problem}
Пусть $g_{ij}$~--- тензор типа $(0, 2)$ такой, что $\mbox{det}(g_{ij})\ne 0$,
$(g^{ij})=(g_{ij})^{-1}$ ($(g_{ij})$ и $(g^{ij})$~--- соответствующие
матрицы). Доказать, что $g^{ij}$~--- тензор типа $(2, 0)$.
\end{problem}
