\chapter*{Лекция 7.}
\section*{Тема: Аппарат теории экстремума (завершение)}
\subparagraph{1. Лемма о замкнутости образа.}
\begin{lemma}
Пусть $X$, $Y$, $Z$~--- банаховы пространства, $A:X\to Y$, $B:X\to Z$~--- линейные операторы.
Пусть $AX=Y$ и образ $B(Ker\,A)$ (ядра оператора $A$ при отображении оператором $B$) замкнут в
$Z$. Тогда образ $TX$ пространства $X$ при отображении оператором $T:X\to Y\times Z$
таким, что $Tx=(Ax,Bx)$ $\forall\,x\in X$ замкнут в произведении $Y\times Z$.
\end{lemma}
\begin{proof}
Пусть последовательность $\{x_n\}$ в $X$ такова, что
$$
(Ax_n,Bx_n)\to(y,z)\in Y\times Z
$$
сильно сходится, т.\,е.
$$
\|Ax_n-y\|\to0,\quad \|Bx_n-z\|\to0\qquad (n\to\infty).
$$
Требуется доказать, что $\exists\,x\in X$ такой, что $Ax=y, Bx=z$. Тем самым лемма
будет доказана.

Из условия $AX=Y$ следует, что $\exists\,x_y\in X$ такой, что
$Ax_y=y$. Тогда
$$
(A(x_n-x_y),B(x_n-x_y))\to (0,z-Bx_y).
$$
Поскольку $A(x_n-x_y)\to 0$ и $AX=Y$, то по лемме о правом обратном $\exists$
последовательность $\bar{x}_n$ такая, что
$$
A(x_n-x_y)=A\bar{x}_n,\qquad \|\bar{x}_n\|\to 0.
$$
Тогда
$$
A(x_n-x_y-\bar{x}_n)=0\quad \forall\,n,\mbox{\qquad т.\,е.\ }
(x_n-x_y-\bar{x}_n)\in Ker\,A\quad \forall\,n
$$
$$
B(x_n-x_y-\bar{x}_n)\to z-Bx_y\qquad (n\to\infty)
$$
На ядре оператора $A$ оператор $B$ имеет замкнутый образ $\Rightarrow\exists\,x_0
\in Ker\,A$ такой, что
$$
Bx_0=z-Bx_y.
$$
Тогда
$$
B(x_0+x_y)=z
$$
$$
A(x_0+x_y)=Ax_y=y.
$$
Лемма доказана.
\end{proof}
Ясно, что в этой лемме вместо условия $AX=Y$ можно предположить,
что образ $Y_1:=AX$ замкнут в $Y$. Тогда $A:X\xrightarrow{HA}Y_1$
и, следовательно, образ $T$ снова замкнут.

Обычно в оптимальном управлении используется следующее
ут\-верж\-де\-ние, вытекающее из леммы о замкнутости образа.
\begin{sled}
Пусть $A:X\to Y$~--- линейный оператор <<на>>, а \mbox{$B:X\to Z$}~---
конечномерный линейный оператор $(dim\,Z<\infty)$. Тогда оператор $T(x)=(Ax,Bx)$ имеет
замкнутый образ в $Y\times Z$.
\end{sled}
\begin{proof}
Действительно, $dim\,B(Ker\,A)<\infty$ $\Rightarrow$ $B(Ker\,A)$~--- замкнутое
подпространство в $Z$.
\end{proof}
\subparagraph{2. Лемма об аннуляторе ядра линейного оператора.}
(АТФ, стр.\,130)
\begin{lemma}
Пусть $A:X\to Y$~--- линейный оператор <<на>>. Тогда
$$
(Ker\,A)^*=A^*Y^*,
$$
где $A^*$~---
сопряженный оператор.
\end{lemma}
\begin{proof}
\begin{enumerate}
\item[а)] $A^*Y^*\subset(Ker\,A)^*$. Действительно, пусть \mbox{$x^*=A^*y^*$}, т.\,е.
$$
\langle x^*,x\rangle=\langle y^*,Ax\rangle\qquad\forall\,x\in X.
$$
Тогда
$$
\langle x^*,x\rangle=\langle y^*,Ax\rangle=\langle y^*,0\rangle=0\qquad\forall\,x\in Ker\,A,
$$
т.\,е. $x^*\in Ker\,A^*$.
Здесь условие $AX=Y$ не требуется.
\item[б)] Покажем, что $(Ker\,A)^*\subset A^*Y^*$. Пусть $x^*\in(Ker\,A)^*$, т.\,е.
$$
\langle x^*,x\rangle=0\qquad\forall\,x\in Ker\,A.
$$
Рассмотрим оператор
$$
Tx=(Ax,\langle x^*,x\rangle),
$$
отображающий $X$ в произведение $Y\times\mathbb{R}$. Согласно следствию из
леммы о замкнутости образа, образ $TX$ замкнут в $Y\times\mathbb{R}$. При этом
$$
TX\neq Y\times\mathbb{R}
$$
поскольку, например, $(0,1)\notin TX$, ибо если $Ax=0$, то и $\langle x^*,x\rangle=0$.
Тогда по лемме о нетривиальности аннулятора на произведении $Y\times\mathbb{R}$ имеется
ненулевой функционал равный нулю на образе $TX$, т.\,е. $\exists\,y^*\in Y^*$ и
$c\in \mathbb{R}^1$ такие, что
$$
\langle y^*,Ax\rangle+c\langle x^*,x\rangle=0\qquad \forall\,x\in X
$$
и при этом
$$
\|y^*\|+|c|>0
$$
Если $c=0$, то $\langle y^*,Ax\rangle=0\ \forall\,x\in X
\Rightarrow \langle y^*,y\rangle=0\ \forall\,y\in Y$ (ибо $AX=Y$)
$\Rightarrow y^*=0$.

Значит $c\neq0$. Тогда
$$
\langle x^*,x\rangle=-\langle \frac{1}{c}y^*,Ax\rangle\qquad\forall\,x\in X.
$$
Таким образом,
$$
x^*\in A^*Y^*.
$$
Лемма доказана.
\end{enumerate}
\end{proof}

\subparagraph{3. Производная по направлению. Производная Гато.} (АТФ, стр.\,137)

Пусть $X$, $Y$~--- банаховы пространства, $U\subset X$~--- открытое множество,
$x_0\in U$~--- точка, $F:U\to Y$~--- оператор, $h\in X$~--- фиксированный
элемент (направление).
\begin{defin}
Пусть существует предел
$$
\lim_{\varepsilon\to+0}\frac{F(x_0+\varepsilon h)-F(x_0)}{\varepsilon}.
$$
Тогда он называется \emph{производной} оператора $F$ в точке $x_0$ \emph{по напрвлению}
$h$ и обозначается $F'(x_0;h)$.
\end{defin}
\begin{defin}
Пусть оператор $F$ в точке $x_0$ имеет производную $F'(x_0,h)$ по любому
направлению $h\in X$, и пусть отображение
$$
h\in X\mapsto F'(x_0,h)\in Y
$$
представляет собой линейный непрерывный оператор. Обозначим его $F'_\Gamma(x_0)$.
Он переводит произвольный элемент $h\in Y$ в элемент
$$
F'_\Gamma(x_0)h=F'(x_0,h)\in Y,
$$
и называется \emph{производной Гато} оператора $F$ в точке $x_0$.
\end{defin}
   Если оператор \emph{дифференцируем по Гато} в точке $x_0$, то есть имеет в этой точке
производную Гато, то он имеет производную по любому направлению $h$, которая дается
формулой
$$
F'(x_0,h)=F'_\Gamma(x_0)h.
$$
Далее, если оператор имеет в точке $x_0$ производную Фреше $F'(x_0)$, то он имеет
и производную Гато $F'_\Gamma(x_0)$, и они совпадают
$$
F'_\Gamma(x_0)=F'(x_0).
$$
Обратное не имеет места, т.\,к., например, уже в $\mathbb{R}^2$
функция диф\-фе\-рен\-ци\-руе\-мая по Гато в точке не обязана быть
непрерывной в этой точке (см. АТФ стр.\,141), в то время как из
дифференцируемости по Фреше непрерывность, конечно, следует.

Итак, для точки $x_0$ мы имеем:
$$
\text{непрерывная дифференцируемость}
$$
$$
\Downarrow
$$
$$
\text{строгая дифференцируемость}
$$
$$
\Downarrow
$$
$$
\text{дифференцируемость по Фреше}
$$
$$
\Downarrow
$$
$$
\text{дифференцируемость по Гато}
$$
$$
\Downarrow
$$
$$
\text{дифференцируемость по каждому направлению.}
$$

Отметим также, что строгая дифференцируемость оператора $g$ в
точ\-ке $x_0$ (а значит, и непрерывня дифференцируемость в точке)
влечет лип\-ши\-це\-вость в окрестности этой точки. Это вытекает,
например, из пред\-став\-ле\-ния
$$
g(x)=g'(x_0)x+(g(x)-g'(x_0)x),
$$
где $(g(x)-g'(x_0)x)$ имеет произвольную малую константу Липшица, если окрестность
точки $x_0$ достаточно мала (см. Лекцию 5). Таким образом, константа Липшица
оператора $g$ в окрестности точки $x_0$ сколь угодно близка к величине $\|g'(x_0)\|$, если
окрестность точки достаточно мала.

\section*{Тема: Условия минимума в абстрактных прост\-ранст\-вах.}
\subparagraph{1. Минимизация функционала на множестве.}

Пусть $X$~--- ба\-на\-хо\-во пространство, $f:X\to \mathbb{R}$~---
функционал, $M\subset X$~--- множество, $x_0\in X$~--- точка.
Ниже достаточно считать, что $f$ определен в некоторой
окрестности $U$ точки $x_0$.

\begin{defin}
$x_0$~--- точка \emph{локального минимума $f$ на $M$}, если $\exists\,\varepsilon>0$
такое, что $f(x)\geqslant f(x_0)\ \forall\,x\in M$ таких, что $\|x-x_0\|<\varepsilon$.
\end{defin}

Нас будут интересовать необходимые условия локального минимума~$f$ на $M$. Чтобы
их получить, нужно предположить определенные свойства $f$ в окрестности точки $x_0$.

\begin{theorema}
Пусть $f$ имеет производную в точке $x_0\in M$ по любому направлению $\bar{x}\in TM(x_0)$
и пусть $f$ удовлетворяет условию Липшица в окрестности точки $x_0$. Пусть $x_0$~---
точка локального минимума $f$ на $M$. Тогда
$$
f'(x_0,\bar{x})\geqslant0\qquad \forall\,\bar{x}\in TM(x_0).
$$
\end{theorema}
\begin{proof}
Пусть выполнены условия теоремы и пусть $\bar{x}\in TM(x_0)$. Найдется
$\tilde{x}(\varepsilon):(0,\varepsilon_0)\to X\ (\varepsilon_0>0)$ такая, что при
$\varepsilon\in(0,\varepsilon_0)$
$$
x_0+\varepsilon\bar{x}+\tilde{x}(\varepsilon)\in M,\quad
\|\tilde{x}(\varepsilon)\|=o(\varepsilon).
$$

Из наличия локального минимума в $x_0$ на $M$ вытекает, что
$$
f(x_0+\varepsilon\bar{x}+\tilde{x}(\varepsilon))\geqslant f(x_0)
$$
при всех достаточно малых $\varepsilon>0$. Поскольку $f$
удовлетворяет ус\-ло\-вию Липшица в окрестности точки $x_0$ с
некоторой константой $L>0$, то
$$
f(x_0+\varepsilon\bar{x}+\tilde{x}(\varepsilon))\leqslant f(x_0+
\varepsilon\bar{x})+L\|\tilde{x}(\varepsilon)\|=f(x_0+\varepsilon\bar{x})+o(\varepsilon).
$$
Следовательно,
$$
f(x_0)\leqslant f(x_0+\varepsilon\bar{x})+o(\varepsilon).
$$
Отсюда вытекает, что
$$
\lim_{\varepsilon\to +0}\frac{f(x_0+\varepsilon\bar{x})-f(x_0)}{\varepsilon}\geqslant0,
$$
т.\,е.
$$
f'(x_0,\bar{x})\geqslant 0.
$$
\end{proof}
\begin{theorema}
Пусть $f$ строго дифференцируема в точке $x_0\in M$, и пусть
$TM(x_0)$~--- подпространство. Тогда необходимое условие
локального ми\-ни\-му\-ма функции $f$ на $M$ в $x_0$ состоит в
том, что
$$
f'(x_0)\bar{x}=0\qquad\forall\,\bar{x}\in TM(x_0).
$$
\end{theorema}
\begin{proof}
Пусть $\bar{x}\in TM(x_0)$~--- произвольный элемент. Поскольку $TM(x_0)$~---
подпространство , то и $(-\bar{x})\in TM(x_0)$. По теореме~1
\begin{equation*}
f'(x_0)\bar{x}\geqslant 0\quad\text{и}\quad f'(x_0)(-\bar{x})\geqslant 0
\end{equation*}
Следовательно, $f'(x_0)\bar{x}=0$.
\end{proof}
\begin{exam}
В простейшей задаче вариационного исчисления
$$
{\cal{J}}(x)=\int_{t_0}^{t_1}F(t,x,\dot{x})\,dt\to \min,
$$
$$
x(t_0)=a,\quad x(t_1)=b,
$$
рассматриваемой в пространстве $C^1$ (или $\mathbb{W}_1^1$),
функционал $\cal{J}$ непрерывно дифференцируем, а множество $M$
определено граничными условиями. Ясно, что $TM(x^0)$ в любой
точке допустимой точке $x^0$ есть под\-прост\-ранст\-во,
определенное условиями:
$$
\bar{x}(t_0)=\bar{x}(t_1)=0.
$$
Необходимое условие локального (т.\,е. слабого) минимума в точке $x^0$ состоит в том,
что на этом подпространстве производная Фреше функционала (т.\,е. первая вариация)
$$
\int_{t_0}^{t_1}F'_x\bar{x}+F'_{\dot{x}}\dot{\bar{x}}\,dt.
$$
равна нулю (где $F_x$ и $F_{\dot{x}}$ вычисляются вдоль траектории $(t,x^0(t),\dot{x}^0(t))$.
Отсюда, как мы знаем, вытекает уравнение Эйлера.
\end{exam}
