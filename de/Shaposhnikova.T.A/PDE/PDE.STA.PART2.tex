\section{Гармонические функции, их свойства}

\begin{df}
Пусть ${u\in C^2(\Omega),\ \Omega\subset\mathbb(R)^n}$ - область.
Функция u называется \emph{гармонической} в $\Omega$, если $\forall x\in\Omega\ \Delta u = 0$.
\end{df}

{\bf Примеры}
\begin{enumerate}
\item $u=\sum_{i=1}^n a_ix_i+b$ - гармоническая в $\mathbb{R}^n$.
\item $n=3:\ u=\frac{1}{r},\ r=|x-x_0|$ - гармоническая при $x\neq x_0$.
\item $n=2:\ \ln{r},\ r^{\pm m} \cos{m\varphi},\ r^{\pm m} \sin{m\varphi}$ -  гармонические.
\end{enumerate}

Пусть теперь $u,v\in C^2(\overline{\Omega}),\ \Omega \ \text{ограниченная область в}\ \
\mathbb{R}^n,\ \partial\Omega\in C^1$ тогда вспомним, что

$$
\int\limits_{\Omega}uv_{x_j x_j}dx =
\int\limits_{\Omega}\frac{\partial u}{\partial x_j}\frac{\partial v}{\partial x_j}dx+
\int\limits_{\partial \Omega} u\frac{\partial v}{\partial x_j} \nu_j dS
$$
$$
\int\limits_{\Omega} divA dx= \int\limits_{\partial \Omega}(A,\nu)dS :\
A=(0,\dots,u\frac{\partial v}{\partial x_j},\dots,0)
$$
\subsection{Формулы Грина}
\begin{stm}[Первая формула Грина] \label{sec1:stm1}
\begin{equation} \label{sec1:eq1}
\int\limits_{\Omega} u\Delta v dx=-
\int\limits_{\Omega}(\nabla u,\nabla v) dx+
\int\limits_{\partial \Omega} u \frac{\partial v}{\partial \nu}dS,
\end{equation}
где $\nu$~--- единичная внешняя нормаль к границе.
\end{stm}

\begin{stm}[Вторая формула Грина] \label{sec1:stm2}
\begin{equation} \label{sec1:eq2}
\int\limits_{\Omega} (u\Delta v -v\Delta u) dx=
\int\limits_{\partial \Omega}\left(u \frac{\partial v}{\partial \nu} - 
v \frac{\partial u}{\partial \nu}\right)dS.
\end{equation}
\end{stm}

\subsection{Фундаментальное решение оператора Лапласа}
Рассмотрим гармонические функции, зависящие только от расстояния до точки, т.е
гармонические функции вида:
$$
v(x) = v(|x-x_0|), \, x\neq x_0, \, \Delta v=0.
$$
Найдём такие гармонические функции $v$, что для любых 
финитных функций~$\phi$ в~области $\Omega$ имеет есто тождество
\begin{equation} \label{sec1:eq3}
\int\limits_{\mathbb{R}^n} v(|x-x_0|)\Delta\varphi(x)dx = \varphi(x_0).
\end{equation}
Положим $r := |x - x_0|$. Тогда оператор Лапласа переписывается как
$$
\Delta v(r) = v''+\frac{n-1}{2} v'.
$$
Учитывая гармоничность функции $v$, получаем
\begin{equation} \label{sec1:eq4}
v(|x-x_0|)=C_1|x-x_0|^{2-n}+C_2, \, n \geq 3, \qquad
v(|x-x_0|)=C_1\ln{|x-x_0|}+C_2, \, n = 2.
\end{equation}
Рассмотрим левую часть равенства \eqref{sec1:eq3}.
Для $n \geq 3$ в силу гармоничности $v$ она равна
\begin{equation} \label{sec1:eq5}
\lim\limits_{\varepsilon\rightarrow 0}
\int\limits_{\mathbb{R}^n \setminus \overline{T}_\varepsilon^{x_0}}
v(|x-x_0|)\Delta \varphi dx =
\lim\limits_{\varepsilon\rightarrow 0}
\int\limits_{\mathbb{R}^n \setminus \overline{T}_\varepsilon^{x_0}}
(v(|x-x_0|)\Delta\varphi - \varphi\Delta v(|x-x_0|))dx.
\end{equation}
К правой части выражения \eqref{sec1:eq5} применим 
вторую формулу Грина \eqref{sec1:eq2}, тогда
\begin{equation} \label{sec1:eq6}
\lim\limits_{\varepsilon\rightarrow 0}
\int\limits_{\mathbb{R}^n \setminus \overline{T}_\varepsilon^{x_0}}
v(|x-x_0|)\Delta \varphi dx =
\lim\limits_{\varepsilon\rightarrow 0}
\int\limits_{\partial \Omega \cup S_\varepsilon^{x_0}}
\left(
v(|x-x_0|) \partial_{\nu} \ph
-\varphi (x) \partial_{\nu} v(|x-x_0|)
\right) dS.
\end{equation}
В силу финитности $\ph$ интеграл по $\partial \Omega$ равен нулю.
Положим $K := \max_{\Omega} \partial_{\nu} \ph$,
$\omega_n$~--- площадь поверхности $n$-мерного шара.
Заметим, что имеет место оценка
$$
\left|\int_{S_{\ep}^{x_0}} |x - x_0|^{2-n} \partial_{\nu} \ph dS\right| \leq
\ep^{2-n} \omega_n K \ep^{n-1} = \omega_n K \ep.
$$
Следовательно, в силу представления \eqref{sec1:eq4} для $v$ получаем
$$
\lim\limits_{\varepsilon\rightarrow 0}
\int\limits_{S_\varepsilon^{x_0}} v(|x - x_0|) \partial_{\nu} \ph dS=0.
$$
Далее заметим, что
$$
C_1\frac{\partial}{\partial\nu}r^{2-n}=-C_1\frac{\partial}{\partial r}r^{2-n}=
-(2-n)C_1r^{1-n}.
$$
Подставляя в выражение \eqref{sec1:eq6} полученные результаты,
приходим к
\begin{multline*}
\lim\limits_{\varepsilon\rightarrow 0}
\int\limits_{\mathbb{R}^n \setminus \overline{T}_\varepsilon^{x_0}}
v(|x-x_0|)\Delta \varphi dx = 
-C_1(n-2)
\lim\limits_{\varepsilon\rightarrow 0}
\int\limits_{S_\varepsilon^{x_0}}
\varphi (x) dS=
-C_1(n-2)
\lim\limits_{\varepsilon\rightarrow 0}
\varphi(x_\epsilon)\omega_n\varepsilon^{n-1}\varepsilon^{1-n} = \\
=-C_1(n-2)\omega_n\varphi(x_0).
\end{multline*}
Учитывая представление \eqref{sec1:eq4}, получаем
$$
C_1=\frac{-1}{\omega_n(n-2)},\, n \ge 3, \qquad
C_1=\frac{1}{2\pi}, \, n=2.
$$

\begin{df}
\emph{Фундаментальное решение оператора Лапласа.}
$$
E(x,x_0)=
\begin{cases}
-\frac{|x-x_0|^{2-n}}{\omega_n(n-2)},\ n\ge 3\\
\frac{1}{2\pi}\ln{(x-x_0)},\ n=2
\end{cases}
$$
\end{df}
Основное характерное свойство фундаментального решения:
$$
\int\limits_{\mathbb{R}^n}E(x,x_0)\Delta\varphi(x)dx=
\varphi(x_0), \, \forall\varphi\in C_0^\infty(\Omega).
$$
На языке обобщенных функций это выглядит так
$$
(\Delta E(x,x_0),\varphi (x))=\varphi(x_0)=(\delta(x-x_0),\varphi), \qquad
\Delta E(x,x_0)=\delta(x-x_0).
$$

\subsection{Представление в виде суммы трёх потенциалов}
Пусть $\Omega$ -- ограниченная область в $\R^n$,
$\partial \Omega$ -- липшецева, ${u \in C^2(\overline{\Omega})}$, ${x_0 \in \Omega}$.
Положим ${\Omega_{\ep} := \Omega \setminus \overline{T}_{\ep}^{x_0}}$.
В силу соотношения \eqref{sec1:eq4} справедливо равенство
\begin{equation} \label{sec1:eq7}
\Delta_x E(x, x_0) = 0, \, x \in \Omega_{\ep}.
\end{equation}

По второй формуле Грина \ref{sec1:stm2}
\begin{equation} \label{sec1:eq8}
\int_{\Omega_{\ep}} \left(u \Delta E - \Delta u E\right) dx 
= \int_{\partial \Omega} \left(u \partial_{\nu} E - \partial_{\nu} u E\right) dS 
+ \int_{\partial T_{\ep}^{x_0}} \left(u \partial_{\nu} E - \partial_{\nu} u E\right) dS.
\end{equation}
В равенстве \eqref{sec1:eq8} устремим $\ep$ к нулю, 
в силу тождества \eqref{sec1:eq7} получим
$$
- \int_{\Omega} \Delta u E dx 
= \int_{\partial \Omega} (u \partial_{\nu} E - \partial_{\nu} u E) dS 
+ \lim\limits_{\ep \to 0} \int_{\partial T_{\ep}^{x_0}} u \partial_{\nu} E dS 
= \int_{\partial \Omega} (u \partial_{\nu} E - \partial_{\nu} u E) dS - u(x_0).
$$

Таким образом, получаем, что
\begin{equation} \label{sec1:eq9}
u(x_0) =
\int\limits_{\Omega}E(x,x_0)\Delta u dx
+ \int\limits_{\partial \Omega} u \partial_{\nu} E(x,x_0) dS
- \int\limits_{\partial \Omega} E(x,x_0) \partial_{\nu} u dS.
\end{equation}
Каждое слагаемое в полученном представении имеет своё название.
\begin{df}
Выражение 
$$
P_1(x_0) := \int_{\Omega} E(x,x_0) \Delta u dx
$$
называется \emph{объёмным потенциалом}.

Выражение 
$$
P_2(x_0) := \int_{\partial\Omega} u \partial_{\nu} E(x,x_0) dS
$$
называется \emph{потенциалом двойного слоя}.

Выражение 
$$
P_3(x_0) := \int_{\partial\Omega} E(x,x_0) \partial_{\nu} u dS
$$
называется \emph{потенциалом простого слоя}.
\end{df}

\subsection{Теоремы о среднем для гармонических функций}
\begin{theorem}[О среднем по сфере] \label{sec1:theorem1}
Пусть $u\in C^2(T_R^{x_0})\cap C(\overline{T}_R^{x_0})$, предположим, что $\Delta u=0$ в $T_R^{x_0}$
тогда
$$
u(x_0)=\frac{1}{|S_R^{x_0}|}\int_{S_R^{x_0}} u(x)dS
$$
\end{theorem}

\begin{proof}
Рассмотрим $T_\rho^{x_0}$, ${\rho < R}$. Тогда в силу представления в виде
суммы трёх потенциалов \eqref{sec1:eq9}, где один из них равен нулю в силу гармоничности, имеем
$$
u(x_0)=
\int_{S_\rho^{x_0}} u \partial_{\nu} E(x,x_0) dS
- \int_{S_\rho^{x_0}} E(x,x_0) \partial_{\nu} u dS.
=
\frac{1}{\omega_n\rho^{n-1}}\int\limits_{S_\rho^{x_0}} u(x)dS
$$
Заметим, что по формуле Гаусса - Остроградского
$$
\int_{S_\rho^{x_0}} E(x,x_0) \partial_{\nu} u dS 
= \int_{S_\rho^{x_0}} \frac{\rho^{2-n}}{(2-n) \omega_n} \partial_{\nu} u dS =
\frac{\rho^{2-n}}{(2-n) \omega_n} \int_{T_{\rho}^{x_0}} \Div \nabla u dx = 0.
$$
Из того, что
$$
\partial_{\nu} E(x, x_0) = \frac{1}{\omega_n \rho^{n-1}} = \frac{1}{|S_{\rho}^{x_0}|},
$$
следует
$$
u(x_0)= \frac{1}{|S_{\rho}^{x_0}|}
\int_{S_\rho^{x_0}} u dS.
$$
Переходя к пределу при $\rho \to R$, получаем искомое.
\end{proof}

\begin{theorem}[О среднем по шару] \label{sec1:theorem2}
Пусть $u \in C^2(T_R^{x_0}) \cap C(\overline{T}_R^{x_0})$. 
Предположим, что $\Delta u=0$ в $T_R^{x_0}$.
Тогда
$$
u(x_0)=\frac{1}{|T_R^{x_0}|}\int_{T_R^{x_0}} u(x)dS.
$$
\end{theorem}

\begin{proof}
Из теоремы о среднем по сфере \ref{sec1:theorem1} следует,
что для $\rho < R$
\begin{equation} \label{sec1:eq10}
\omega_n\rho^{n-1}u(x_0)=\int\limits_{S_{\rho}^{x_0}}u(x)dS.
\end{equation}
В силу теоремы Фубини получаем
$$
|T_R^{x_0}| u(x_0) 
= \frac{\omega_n}{n} R^n u(x_0) 
= \int_0^R \omega_n \rho^{n-1} u(x_0) u(x_0) d \rho
= \int_0^R \int\limits_{S_{\rho}^{x_0}}u(x)dS d \rho
= \int_{T_R^{x_0}} u(x)dS.
$$
\end{proof}

\begin{theorem} \label{sec1:theorem3}
Пусть $\ph \in C[0, R]$. Положим
$$
A_{\ph} := \int_0^R \ph(|x - x_0|) dx.
$$
Прдположим, что $A_{\ph}$ отлично от нуля.
Пусть $u \in C^2(T_{R}^{x_0}) \cap C(\overline{T}_R^{x_0})$.
Тогда
$$
u(x_0) = \frac{1}{A_{\ph}} \int_{T_R^{x_0}} u(x) \ph(|x - x_0|) dx.
$$
\end{theorem}

\begin{proof}
Умножим тождество \eqref{sec1:eq10} обоих сторон на 
${\ph(\rho) = \ph(|x - x_0|)}$ и проинтегрируем по $\rho$.
$$
u(x_0) \int_0^R \omega_n \rho^{n-1} \ph(\rho) d \rho
= \int_0^R \int_{S_{\rho}^{x_0}} \ph(|x - x_0|) dS dx
= \int_{T_{R}^{x_0}} \ph(|x - x_0|) dS dx.
$$
\end{proof}

\begin{theorem}[О бесконечной дифференцируемости]
Пусть $\Omega$ область в $\mathbb{R}^n$, $u(x)$ гармоническая в $\Omega$.
Тогда $u \in C^\infty(\Omega)$.
\end{theorem}

\begin{proof}
Пусть $x \in \Omega$, $h > 0$, $\overline{T}_{h}^x \subset \Omega$,
$\omega_h$ -- ядро усреднения.
Рассмотрим
$$
u_h(x) = \int_{\R_y^n}\omega_h(|x-y|)u(y)dy
= \int_{|x-y|<h}\omega_h(|x-y|)u(y)dy
= \int_0^h \omega_h(\rho) \int_{S_\rho^x}u(y)dSd\rho.
$$
В силу теоремы о среднем $\int_{S_\rho^x} u(y) dS = u(x) |S_{\rho}^x|$,
поэтому
$$
u_h(x) = u(x) \int_0^h \omega_h(\rho) |S_{\rho}^x| d\rho
= u(x) \int_0^h \int_{S_{\rho}^x} \omega_h(|x - y|) dS_y d\rho
$$
В силу свойств ядра усреднения $\int_0^h \int_{S_{\rho}^x} \omega_h(|x - y|) dS_y d\rho = 1$,
откуда для достаточно малых $h$ имеет место тождество ${u_h(x) = u(x)}$.
Так как функция $u_h(x)$ является бесконечно дифференцируемой.
\end{proof}

\subsection{Принцип максимума}
\begin{theorem}[Слабый принцип максимума] \label{sec1:theorem4}
Пусть $\Omega$ -- ограниченная область в $\R^n$, 
$u \in C^2(\Omega) \cap C(\overline{\Omega})$,
$\Delta u \geq 0$ в $\Omega$. 
Тогда справедливо тождество
$$
\max\limits_{\overline{\Omega}} u 
= \max\limits_{\partial \Omega} u.
$$
\end{theorem}

\begin{proof}
Предположим сначала, что $\Delta u > 0$ в $\Omega$.
Так как функция $u$ непрерывна в $\overline{\Omega}$,
то её максимум достигается в какой-то точке $x_0$.

Пусть $x_0 \in \Omega$. Тогда матрица Якоби $J_u$ является нулевой в этой точке,
и квадратичная форма матрицы вторых частных производных неположительна, то есть
$$
\sum_{i, j = 1}^n \partial_{x_i x_j} u(x_0) \xi_i \xi_j \leq 0.
$$
В частности, $\partial_{x_i x_i} u(x_0) \leq 0$, откуда $\Delta u(x_0) \leq 0$.
Противоречие с предположением, что $\Delta u > 0$ в $\Omega$.
Поэтому в этом случае максимум достигается в точке $x_0$, лежащей на $\partial \Omega$.

Рассмотрим общий случай, когда $\Delta u \geq 0$.
Введём новую функцию
$$
v_{\ep}(x) := u(x) + \ep l(x)
= u(x) + \ep \sum_{i=1}^n x_i^2.
$$
Тогда
$$
\Delta v_{\ep} = \Delta u + n \ep \Delta x_i^2
= \Delta u + 2 n \ep > 0.
$$
Так как $l$ является непрерывной функцией на компакте $\overline{\Omega}$,
то её значения ограничены по модулю какой-то константой $K$. Тогда
$$
\max_{\overline{\Omega}} u \leq \max_{\overline{\Omega}} v_{\ep}
= \max_{\partial \Omega} v_{\ep} \leq \max_{\partial \Omega} u + n K \ep.
$$
Устремляя $\ep$ к нулю, получаем искомое.
\end{proof}

\begin{theorem}[Строгий принцип максимума] \label{sec1:theorem5}
Пусть $u\in C^2(\Omega)\cap C(\overline{\Omega})$, 
$\Omega$ -- ограниченная область с гладкой границей.
Предположим, что $\Delta u = 0$ в $\Omega$.
Положим
$$
\max\limits_{\overline{\Omega}}u(x) := M, \qquad 
\min\limits_{\overline{\Omega}}u(x) := m.
$$
Тогда, если для $x_0 \in \Omega$ справедливо тождество $u(x_0) = M$ ($u(x_0) = m$), 
то $u(x) \equiv \const$.
\end{theorem}

\begin{proof}
Рассмотрим случай, когда в некоторой точке $x_0 \in \Omega$ 
достигается значение $M$ функции $u$. Положим
$$
L := \{x \in \Omega \mid u(x) = M\}.
$$
Докажем, что множество $L$ является открыто-замкнутым подмножеством $\Omega$.
Оно непусто, потому что $x_0 \in L$. 
Замкнутость следует из непрерывности $u$.

Докажем открытость. От противного, пусть существует $x_1 \in L$ такой,
что любая его окрестность в $\Omega$ содержит точку не из $L$.
Выберем такую окрестность $T_{\ep_1}^{x_1}$, что
$\overline{T}_{\ep_1}^{x_1} \subset \Omega$.
Тогда существует $x_2 \in T_{\ep_1}^{x_1}$ такой, 
что $u(x_2) < M$.
В силу непрерывности функции $u$ существует окрестность $T_{\ep_2}^{x_2}$,
$\overline{T}_{\ep_2}^{x_2} \subset T_{\ep_1}^{x_1}$
точки $x_2$ такая, что для любого $x \in T_{\ep_2}^{x_2}$ справедливо
неравенство $u(x) < M - \delta$.
Тогда в силу теоремы о среднем по шару \ref{sec1:theorem2} имеет место оценка
\begin{multline*}
u(x_1) = \frac{1}{|T_{\ep_1}^{x_1}|} \int_{T_{\ep_1}^{x_1}} u(x) dx
= \frac{1}{|T_{\ep_1}^{x_1}|} 
\left(\int_{T_{\ep_1}^{x_1} \setminus \overline{T}_{\ep_2}^{x_2}} u(x) dx 
+ \int_{T_{\ep_2}^{x_2}} u(x) dx\right) \leq \\
\leq \frac{M(|T_{\ep_1}^{x_1}| - |T_{\ep_2}^{x_2}|) 
+ (M - \delta) |T_{\ep_2}^{x_2}|}{|T_{\ep_1}^{x_1}|} 
= M - \delta \frac{|T_{\ep_2}^{x_2}|}{|T_{\ep_1}^{x_1}|} < M.
\end{multline*}
Но так как $x_1 \in L$, то $u(x_1) = M$. Противоречие с не открытостью $L$.
По теореме об открыто-замкнутом подмножестве области $L = \Omega$, 
то есть $u \equiv M$. Для рассмотрения минимума достаточно рассмотреть $v := -u$.
\end{proof}

\begin{imp} \label{sec1:imp1}

\begin{theorem}[Лиувилль]
Пусть $u(x)$ - гармоническая в $\mathbb R^n$, неотрицательная
функция. Тогда $u=const$.
\end{theorem}

\textbf{Доказательство:} Зафиксируем точку $x_0$ и шар $Q_R^{x_0}$
радиуса $R$ с центром в нашей точке.Поскольку производная
гармонической функции -- также функция гармоническая, по теореме о
среднем имеем:
$$
\frac {\partial u}{\partial x_j}(x_0)=\frac 1{|Q_R^{x_0}|}
\int_{Q_R^{x_0}} \frac {\partial u}{\partial x_j}(x)\, dx= \frac
1{|Q_R^{x_0}|} \int_{S_R^{x_0}} u(x) \nu_j(x) dS
$$
Мы использовали формулу Стокса, чтобы перейти к интегрированию по
границе шара. А теперь используем еще одну теорему о среднем, на
этот раз из курса математического анализа:
$$
=\frac 1{|_R^{x_0}|} \nu_j(\tilde{x}) \int_{S_R^{x_0}} u(x)
 dS=\frac{|S_R^{x_0}|}{|Q_R^{x_0}|} \nu_j(\tilde{x}) |u(x_0)|
$$
Здесь $|S_R^{x_0}|=w_n R^{n-1}$,а $|Q_R^{x_0}|=\frac {w_n} n R^n$.
Таким образом,
$$
|\frac {\partial u}{\partial x_j}(x_0)|\leq \frac n R
|u(x_0)|\rightarrow 0 \quad R\rightarrow\infty
$$
Ничего не мешает нам выбрать радиус шара сколь угодно большим, а
значит, $|\frac {\partial u}{\partial x_j}(x_0)|=0\quad \forall
j=1,\ldots,n$. Следовательно, $u=const$, что и требовалось.\\

\textbf{Задача:} Пусть $u(x)$ - гармоническая в $\mathbb R^n$ и
$u(x)\geq -C(1+|x|^m)$, где $c,m>0$ $u(x)$. Показать, что в этом
случае $u(x)$ есть полином степени не выше $[m]$.

Рассмотрим задачу Дирихле для оператора Лапласа:
$$
 \left\{
\begin{aligned}
&\Delta u(x)=0\quad x\in\Omega\\
&u(x)=\varphi(x)\quad x\in \partial\Omega,\,f\in C(\partial\Omega)\\
\end{aligned}
\right.
$$
Под классическим решением понимается решение из класса
$C^2(\Omega)\bigcap C(\overline \Omega)$.

Для такого решения ранее была получена формула
$$
u(x_0)=\int_\Omega E(|x-x_0|) \Delta u(x)\, dx
+\int_{\partial\Omega}u(x) \frac {\partial
E(|x-x_0|)}{\partial\nu}\, dS-\int_{\partial\Omega} E(|x-x_0|)
\frac {\partial u}{\partial \nu}(x)\, dS
$$

Пусть существует функция $g(x,x_0)$ со следующими свойствами:\\
$g(x,x_0)\in C^2_x(\overline \Omega)\,$,  $\Delta_x
g(x,x_0)=0\,\forall x_0 \in \Omega$ и при этом $g(x,x_0)|_{x\in
\partial\Omega}=-E(|x-x_0|)|_{x\in
\partial\Omega},\,\forall x_0 \in \Omega$

Запишем вторую формулу Грина для $u(x),g(x,x_0)$:
$$
\int_\Omega (u\Delta g-g\Delta u )\, dx = \int_{\partial\Omega} u
\frac{\partial g}{\partial \nu}-g \frac{\partial u}{\partial \nu}
$$
Поскольку функции $u$ и $g$ -- гармонические в области, то левая
часть формулы обращается в ноль. Теперь прибавим к левой и правой
частям уже упоминавшееся соотношение:
$$
u(x_0)=\int_{\partial\Omega}u(x) \frac {\partial
E(|x-x_0|)}{\partial\nu}\, dS-\int_{\partial\Omega} E(|x-x_0|)
\frac {\partial u}{\partial \nu}(x)\, dS
$$
И поскольку $g(x,x_0)+E(|x-x_0|)=0$ на $\partial\Omega$, получаем:
$$
u(x_0)=\int_{\partial\Omega}u(x)
\frac{\partial(E(|x-x_0|)+g(x,x_0))}{\partial\nu}\, dS
$$

\textbf{Определение:} \emph{Функцией Грина} называется функция
$G(x,x_0)=E(|x-x_0|)+g(x,x_0)$, где функция $g(x,x_0)$ введена
ранее. Точка $x_0$ называется \emph{полюсом}.\\

Итак, если $u\in C^2(\overline\Omega)$ -- решение задачи Дирихле,
то:
$$
u(x_0)=u(x_0)=\int_{\partial\Omega}u(x) \frac{\partial
G(x,x_0)}{\partial\nu}\, dS
$$

%\newtheoremem*{lemma}{Лемма}
\begin{lemma}
$\forall x_1,x_0 \in \Omega \quad G(x_1,x_0)=G(x_0,x_1)$
\end{lemma}
\textbf{Доказательство:} Исходя из определения функции Грина,
$\Delta_x G(x,x_0)=0$ при $\neq x_0$. Вырежем вокруг точек $x$ и
$x_0$ шарики маленького радиуса $\varepsilon$, и то что осталось
обозначим за $\Omega_\varepsilon$:
$$
\Omega_\varepsilon=\Omega \\(\overline{Q_\varepsilon^{x_0}\cup
Q_\varepsilon^{x_1}})
$$
Введем функции $u(x)=G(x,x_0)$, и $v(x)=G(x,x_1)$. В области
$\Omega_\varepsilon$ они гармонические, поэтому по второй формуле
Грина:
$$
0=\int_{\Omega_\varepsilon} (u\Delta v- v\Delta u)\,
dx=\int_{\partial \Omega_\varepsilon} \Bigl(u \frac {\partial
v}{\partial \nu}-v \frac {\partial u}{\partial \nu}\Bigr) \, dS
$$
Поскольку $G(x,x_0)|_{\partial\Omega}=0$ (вспомним, что функция
$g$ определялась условием $g(x,x_0)|_{x\in
\partial\Omega}=-E(|x-x_0|)|_{x\in
\partial\Omega},\,\forall x_0 \in \Omega$ ), то
$u(x)|_{\partial\Omega}=v(x)|_{\partial\Omega}=0$. Поэтому
$$
0=\int_{S_\varepsilon^{x_0}}G(x,x_0)\frac{\partial
G(x,x_1)}{\partial\nu}\,dS+\int_{S_\varepsilon^{x_1}}G(x,x_0)\frac{\partial
G(x,x_1)}{\partial\nu}\,dS
$$
Перейдем к пределу при $\varepsilon\rightarrow 0$,
воспользовавшись тем, что
$$
\frac{\partial E(|x-x_1|)}{\partial\nu}= -\frac{\partial
E}{\partial r}(r)=-\frac 1{w_n r^{n-1}}
$$
Получаем,что
$$
\int_{S_\varepsilon^{x_1}}G(x,x_0)\frac{\partial
G(x,x_1)}{\partial\nu}\,dS=-\frac 1{w_n \varepsilon^{n-1}}
\int_{S_\varepsilon^{x_1}} G(x,x_0)\, dS\rightarrow - G (x_1,x_0)
$$
Аналогично переходя к пределу в другом слагаемом, имеем
$$
0=-G(x_1,x_0)+G(x_0,x_1)
$$
что и требовалось доказать.

\subsection{Потенциал простого слоя}

$$
Где $\mu(x)\in C(\Gamma)$, $\Gamma$ -- замкнутая поверхность
Ляпунова.

\begin{theorem}
Потенциал простого слоя непрерывен в $\mathbb R^n$.
\end{theorem}

\textbf{Доказательство:} Непрерывность может нарушаться только при
переходе через $\Gamma$. Вначале проверим,что потенциал определен
в точках поверхности. Заметим, что $\forall x_0\in\Gamma$:
$$
|P_1(x_0)|\leq\max_\Gamma |\mu(x)| \int_\Gamma \frac
1{|x_0-\xi|^{n-2}}\,dS_\xi = M \Bigl( \int_{\Gamma'}\frac
1{|x_0-\xi|^{n-2}}\,dS_\xi+\int_{\Gamma''}\frac
1{|x_0-\xi|^{n-2}}\,dS_\xi\Bigr)
$$
Где $M=\max_\Gamma |\mu(x)|$, $\Gamma'$-пересечение $\Gamma$ с
маленьким шариком с центром в $x_0$, а $\Gamma''$ -- оставшаяся
часть $\Gamma$. Второй интеграл конечен,а конечность первого
проверяется переходом к системе координат $r^2=\rho^2+\xi_n^2$:
$$
\int_{\Gamma'}\frac 1{|x_0-\xi|^{n-2}}\,dS_\xi=\int_{\Gamma'}\frac
{dS_\xi}{r^{n-2}}=\int_{D(x_0)}\frac{d\xi_1\ldots
d\xi_{n-1}}{\rho^{n-2}\cos (\nu_\xi,\xi_n)}\leq C\int_0^d
\rho^{2-n} \rho^{n-2}\,d\rho=Cd
$$


Непрерывность доказывается аналогично предыдущей теореме:
$$
 |P_1(x)-P_1(y)|\leq |P'_1(x)|+|P'_1(y)|+|P''_1(x)-P''_1(y)|
$$
Где, как обычно, $P'_1(x)$ и $P''_1(x)$ -- соответственно
интегралы по $\Gamma'=\Gamma\cap Q^x_\eta, \quad \eta\ll 1$ и по
$\Gamma''=\Gamma\backslash\Gamma'$. Последнее слагаемое стремится
к нолю ввиду дифференцируемости $P''$,а первые в силу оценки
$$
|P'_1(x)|\leq M\int_{\Gamma'} \frac{dS_\xi}{|x-\xi|^{n-2}} \leq
M\eta
$$
и выбора достаточно маленького $\eta$\\

Покажем, что $\forall x\in \mathbb R^n$ определена (конечна)
нормальная производная
$$
\frac{\partial P_1}{\partial \nu_x}=\int_\Gamma \mu(\xi)
\frac\partial{\partial\nu_\xi}\frac 1{|x-\xi|^{n-2}}\,dS_\xi
$$
Достаточно проверить существование для $x\in\Gamma$. Аналогично
выкладкам для потенциала двойного слоя,
$\frac\partial{\partial\nu_\xi}\frac 1{|x-\xi|^{n-2}}=-\frac{|\cos
(r,\nu_\xi)}{r^{n-1}}$,поэтому
$$
\bigl| \int_\Gamma \mu(\xi) \frac\partial{\partial\nu_\xi}\frac
1{|x-\xi|^{n-2}}\,dS_\xi\Bigr|\leq M\Bigl( \int_{\Gamma'} \frac
{|cos(r,\nu_x)|}{r^{n-1}}\, dS_\xi+\int_{\Gamma''} \frac
{|cos(r,\nu_x)|}{r^{n-1}}\, dS_\xi \Bigl)
$$
Второй интеграл конечен, а первый, в силу сделанной ранее оценки,
$$
\int_{\Gamma'} \frac {|cos(r,\nu_x)|}{r^{n-1}}\, dS_\xi\leq C_1
\int_{D(x)} \frac {r^\alpha}{r^{n-1}}\,d\xi_1\ldots d\xi_n\leq C_2
\int_0^d \rho^{\alpha-n+1}\rho^{n-2}\,d\rho=C_3 d^\alpha
$$
следовательно, нормальная производная определена $\forall x\in
\mathbb R^n$.

\subsection {Метод Ритца}
Возьмем в $H'$  произвольную линейно независимую систему функций
$\phi_1,\ldots,\phi_k,...$, линейная оболочка которой плотна в

E(u)=E(v_k)$. Ищем $v_k=\sum_{j=1}^k c_j^k \phi_j$.ВВедем функцию
$$
F(c_1,\ldots,c_k)=E(\sum_{j=1}^k c_j^k \phi_j)
$$
В точке минимума должны выполняться условия $\frac{\partial
F}{\partial c_j}=0\ \forall j=1,\ldots,k$, что эквивалентно
системе линейных уравнений
$$
\sum_{i=1}^k c_i
(\phi_i,\phi_j)_{H'}+(f,\phi_j)_{L_2(\Omega)}\quad j=1,\ldots,k
$$
Определитель системы представляет собой определитель Грамма для
системы $\phi_1,\ldots,\phi_k$ и не равен нолю в силу их линейной
независимости. Поэтому существует решение $c_1^k,\ldots,c^k_k$, и
элемент $v_k=\sum_{j=1}^k c_j^k \phi_j$, реализующий минимум на
$R_k$.

Последовательность $\{v_k\}$ называется \emph{последовательностью
Ритца}.
\begin{theorem}
Последовательность Ритца является минимизирующей функционал
$E(\cdot)$ на подпространстве $H'$ последовательностью.
\end{theorem}

$$
R_1\subset R_2\subset R_3\subset \ldots

$$
E(v_1)\geq E(v_2)\geq E(v_3) \geq \ldots \geq d

\int_{\omega_\tau}(vTu-\,uT^*v)\,dxdt=\int_{\omega_\tau}(vu_t+uv_t-v\Delta
u+u\Delta v )\,dxdt=
$$
$$
=\int_\Omega u(x,\tau) v(x,\tau)\,dx-\int_\Omega u(x,0)
v(x,0)\,dx+\int_{S_\tau} \bigl(u \frac{\partial v}{\partial \nu}+v
\frac{\partial u}{\partial \nu}\bigr)\,dS
$$\\


\noindent Проверим, что фундаментальным решение для оператора
теплопроводности является
$$
\Gamma(x,x_0,t,t_0)=\frac {\theta(t-t_0)}{(2\sqrt{\pi(t-t_0)})^n}
e^{-\frac{|x-x_0|^2}{4(t-t_0)}}
$$
То есть нужно проверить
$$
\bigr(T_{x,\,t}
\Gamma(x,x_0,t,t_0),f(x,t)\bigl)=\bigr(\Gamma(x,x_0,t,t_0),T^*f(x,t)\bigl)=f(x_0,t_0)
$$
Убедимся, что $\Gamma\in L_{1,\, loc}(\mathbb R^{n+1})$:

$$
\frac 1{2^n \pi ^{n/2}}
\int\limits_{t_0}^{t_0+R}\int\limits_{|x-x_0|<\,C} \frac
1{|t-t_0|^{n/2}}
e^{-\frac{|x-x_0|^2}{4(t-t_0)}}\,dxdt\Biggl|_{\xi=\frac{x-x_0}{2\sqrt{t-t_0}}}=\frac
1{\pi^{n/2}} \int\limits_{t_0}^{t_0+R}\int\limits_{|\xi|<\frac C
{2\sqrt{t-t_0}}} e^{-|\xi|^2}\,d\xi dt\leq
$$
$$
\leq \frac 1{\pi^{n/2}}
\int\limits_{t_0}^{t_0+R}\int\limits_{\mathbb R^n}
e^{-|\xi|^2}\,d\xi dt=R
$$
То есть
$$
\bigr(\Gamma(x,x_0,t,t_0),T^*f(x,t)\bigl)=\int_{\mathbb R^{n+1}}
\Gamma(x,x_0,t,t_0)T^*f(x,t)\,dx dt=
$$
$$
\lim_{\varepsilon\rightarrow 0}
\int\limits_{t_0+\varepsilon}^{\infty}\int\limits_{\mathbb
R^n}\Gamma(x,x_0,t,t_0)T^*f(x,t)\,dx dt=
$$
Пользуясь формулой Грина,
$$
=\lim_{\varepsilon\rightarrow 0}\Bigl(
\int\limits_{t_0+\varepsilon}^{\infty}\int\limits_{\mathbb
R^n}T\Gamma(x,x_0,t,t_0)T f(x,t)\,dx dt+ \int\limits_{\mathbb
R^n}\Gamma(x,x_0,t_0+\varepsilon,t_0)f(x,t_0+\varepsilon)\,dx\Bigl)
$$
\textbf{Упражнение:} Для любого $t>t_0$ $T\Gamma=0$.\\
Используя этот факт, пишем:
$$
=\lim_{\varepsilon\rightarrow 0} \int\limits_{\mathbb
R^n}\Gamma(x,x_0,t_0+\varepsilon,t_0)f(x,t_0+\varepsilon)\,dx=\lim_{\varepsilon\rightarrow
0} \frac 1{(2\sqrt{\pi\varepsilon})^n} \int\limits_{\mathbb R^n}
e^{-\frac{|x-x_0|^2}{4\varepsilon}}
f(x,t_0+\varepsilon)\,dx\Biggl|_{\xi=\frac{x-x_0}{2\sqrt\varepsilon}}=
$$

$$
=\lim_{\varepsilon\rightarrow 0} \frac
1{\pi^{n/2}}\int\limits_{\mathbb
R^n}e^{-|\xi|^2}f(x_0+2\sqrt\varepsilon\xi,t_0+\varepsilon)\,d\xi=
$$
$$
=\lim_{\varepsilon\rightarrow 0} \Bigl\{ \frac
1{\pi^{n/2}}\int\limits_{|\xi|>N}e^{-|\xi|^2}(f(x_0+2\sqrt\varepsilon\xi,t_0+\varepsilon)-f(x_0,t_0))\,d\xi+\frac
1{\pi^{n/2}}
\int\limits_{|\xi|<N}e^{-|\xi|^2}(f(x_0+2\sqrt\varepsilon\xi,t_0+\varepsilon)-f(x_0,t_0))\,d\xi
\Bigr\}
$$
$$
+f(x_0,t_0)
$$
Покажем, что интегралы стремятся к нолю:
$$
M=\sup_{\mathbb R^n} |f(x,t)|<\infty \quad
\forall\varepsilon>0\quad \exists N \quad \frac
M{\pi^{n/2}}\int\limits_{|\xi|>N}e^{-|\xi|^2}\,d\xi<\varepsilon/2
$$
$$
\forall\varepsilon>0\quad \delta>0\quad \forall x,t:
|x-x_0|<\delta,|t-t_0|<\delta\quad |f(x,t)-f(x_0,t_0)|<\varepsilon
$$
Возьмем $2\sqrt\varepsilon N<\delta,\varepsilon<\delta$, тогда и
второй интеграл меньше $\varepsilon$,что мы и хотели показать.
Итак, мы доказали, что $\Gamma(x,x_0,t,t_0)$ является
фундаментальным решением.

\begin{theorem}[Принцип максимума в ограниченной области]
Пусть $u(x,t)$ - решение уравнения $Tu=0$ в слое $\tilde{w}_\tau$,
принадлежащее классу $C^{2,1}(\tilde{w}_\tau)\cap C(\overline{
w_\tau})$.  Тогда $\forall (x,t)\in \tilde{w}_\tau$:
$$
\min_{\sigma_\tau} u(x,t)\leq u(x,t)\leq \max_{\sigma_\tau} u(x,t)
$$
Где $\sigma_\tau=\Omega\cup S_\tau$, $S_\tau$ -- боковая
поверхность цилиндра.
\end{theorem}


\noindent\textbf{Доказательство:}\\
Достаточно доказать, что $\min_{\sigma_\tau} u(x,t)\leq u(x,t)$,
потому что применив это рассуждение к функции $v=-u$, получим
утверждение для максимума.

Из непрерывности $u$ $\exists M>0:\, |u(x,t)|<M$ в $\overline{
w_\tau}$. Выберем $M_1$ такое, что
$$
v(x,t)=u(x,t)-M_1<0 \quad \forall (x,t)\in \overline{ w_\tau}
$$
Тогда $Tv=0$ в $\tilde{w}_\tau$ и $v<0$ в $\overline{ w_\tau}$.
 Положим $v=e^{\gamma t}w,\,\gamma=const>0$. Покажем, что минимум отрицательного значения $w$ может достигаться
лишь на $\sigma_\tau$. Предположим противное: $\exists
(x_0,t_0)\in \tilde{w}_\tau \,: \, 0>w(x_0,t_0)=\min_{\overline{
w_\tau}} w(x,t)$. Ясно, что $w(x,t)<0$ в $\overline{ w_\tau}$,
$\frac{\partial w}{\partial x_j}(x_0,t_0)=0,\ \frac{\partial
w}{\partial t}(x_0,t_0)\leq 0$, а $\frac{\partial^2 w}{\partial
x_j^2}(x_0,t_0)\geq 0$. Но тогда
$$
Tv\Bigl|_{(x_0,t_0)}=\Bigl(\gamma e^{\gamma t}w(x,t)+e^{\gamma
t}\frac{\partial w}{\partial t}(x,t)-e^{\gamma t}\Delta
w(x,t)\Bigr)\Bigl|_{(x_0,t_0)}<0
$$
Что противоречит равенству $Tv=0$.

Следовательно
$$
w(x,t)\geq \min_{\sigma_\tau} w(x,t)
$$
Или, что то же самое,
$$
e^{-\gamma t} v(x,t)\geq \min_{\sigma_\tau} e^{-\gamma t} v(x,t)
$$
Переходя к пределу при $\gamma\rightarrow 0$  получаем $v(x,t)\geq
\min_{\sigma_\tau} v(x,t)$, а отсюда и требуемое неравенство для
$u(x,t)$.

\subsection{Принципы максимума}


\begin{theorem}[Принцип максимума в неограниченной области]
Пусть $u(x,t)$ - решение уравнения $Tu=0$ в слое
$G_\tau=\left\{\,(x,t)\,\Bigl|\,x\in \mathbb R^n,\,0<t\leq \tau\,
\right\}$, принадлежащее классу $C^{2,1}(G_\tau)\cap C(\overline{
G_\tau})$. Предположим, что $\forall (x,t)\in G_\tau\quad
|u(x,t)|\leq M$. Тогда $\forall (x,t)\in G_\tau$:
$$
\inf_{x \in \mathbb R^n} u(x,0)\leq u(x,t)\leq \sup_{x \in \mathbb
R^n} u(x,0)
$$
\end{theorem}


\noindent\textbf{Доказательство:}\\
Обозначим
$$
M_0=\sup_{x \in \mathbb R^n} u(x,0),\,m_0=\inf_{x \in \mathbb R^n}
u(x,0)
$$
Введем вспомогательную функцию $v(x,t)=|x|^2+2nt$. Легко видеть,
что в $G_\tau$ $Tv=v_t-\Delta |x|^2=2n-2n=0$. Введем еще функции
$$
W_1(x,t)=M_0-u(x,t)+\varepsilon v(x,t)
$$
$$
W_2(x,t)=M_0-u(x,t)-\varepsilon v(x,t)
$$
Заметим, что $TW_1=TW_2=0$ в $G_\tau$, и кроме того
$$
W_1(x,0)=M_0-u(x,0)+\varepsilon |x|^2\geq 0
$$
$$
W_2(x,0)=m_0-u(x,0)-\varepsilon |x|^2\leq 0
$$
Далее, $\forall \varepsilon>0\quad \exists R(\varepsilon)\quad
\forall x:\, |x|\geq R(\varepsilon)\quad W_1(x,t)\geq
0,\,W_2(x,t)\leq 0 $ Теперь применим принцип максимума для
ограниченной области  к функциям $W_1$ и $W_2$ в области $G_\tau$:
$$
W_1(x,t)\geq 0,\quad W_2(x,t)\leq 0\qquad (x,t)\in G_\tau
$$
Что равносильно
$$
m_0-\varepsilon v(x,t)\leq u(x,t) \leq M_0+\varepsilon v(x,t)
\qquad \forall(x,t)\in G_\tau
$$
Переходя к пределу при $\varepsilon\rightarrow 0$, получим
утверждение теоремы.\\

\begin{theorem}[строгий принцип максимума ]
Пусть функция $u(x,t)$ в цилиндре
$\tilde{w}_\tau=\left\{\,(x,t)\,\Bigl|\,x\in \Omega,\,0<t\leq
\tau\, \right\}$ удовлетворяет уравнению $Tu=0$, принадлежит
классу $C^{2,1}(\tilde{w}_\tau)\cap C(\overline w_\tau)$  и
принимает в точке $(x_0,t_0)\in \tilde{w}_\tau$ наибольшее
значение, то $u(x,t)\equiv u(x_0,t_0)=const$ в цилиндре
$\tilde{w}_{\tau_0}=\tilde{w}_\tau\cap\{t\leq\tau_0\}$.
\end{theorem}


\noindent\textbf{Доказательство:}\\
Предположим противное. Пусть $u(x_1,t_1)<u(x_0,t_0):=M$, где
$t_1<t_0$. Соединим точки $(x_0,t_0)$ и $(x_1,t_1)$ ломаной,
содержащейся в $\tilde{w}_\tau$ с вершинами в точках
$t_1,\ldots,t_n,t_{n+1}=t_0$, причем $t_1<t_2<\ldots<t_{n+1}=t_0$.
Если мы докажем, что из неравенства $u(x_s,t_s)<M$ следует
$u(x_{s+1},t_{s+1})<M$, то, двигаясь по ломаной, получим
$u(x_0,t_0)<M$ -- противоречие, и теорема будет доказана.

 Пусть
точки $(x_s,t_s)$ и $(x_{s+1},t_{s+1})$ лежат на прямой
$$
x_j=k_j t+a_j,\quad j=1,\ldots,n
$$
Рассмотрим цилиндр $P(x,t)<\rho^2$, где $P(x,t)=\sum_{j=1}^n
(x_j-k_j t-a_j)^2$. Выберем $\rho>0: u(x,t_s)<M-\varepsilon_1\quad
\forall x: P(x,t_s)<\rho^2$. Построим вспомогательную функцию
$$
v(x,t)=e^{-\gamma t}(\rho^2-P(x,t))^2,\quad \gamma>0
$$
Тогда
$$
Tv=e^{-\gamma t}(-\gamma(\rho^2-P)^2-2(\rho^2-P)\frac{\partial
P}{\partial t}+4n(\rho^2-8P)-8P
$$
Покажем, что можно $\gamma$ выбрать настолько большим, что $Tv<0$
внутри цилиндра. Действительно, на боковой поверхности цилиндра
$Tv=-8e^{-\gamma t}\rho^2<0$, поэтому неравенство справедливо и в
некоторой $\delta$-окрестности поверхности. Если же
$P(x,t)<\rho^2-\delta$, то $\rho^2-P(x,t)>\delta$ и при достаточно
большом $\gamma$ первый член в скобках будет больше по модулю, чем
сумма модулей остальных членов, то есть $Tv<0$ внутри цилиндра.
Рассмотрим функцию
$$
W(x,t)=M-u(x,t)-\alpha v(x,t)
$$
Ясно, что $TW=-\alpha Tv>0$ в цилиндре. Тогда по принципу
максимума для ограниченной области,
$$
W(x,t)\geq \min_\sigma W(x,t)
$$
Где $\sigma$ -- "наклонный стакан"(боковая поверхность плюс нижнее
основание нашего цилиндра).На боковой поверхности цилиндра
$W(x,t)=M-u(x,t)>0$, а на нижнем основании
$W(x,t_s)=M-u(x,t_s)-\alpha v(x,t_s)\geq \varepsilon_1-\alpha
v(x_t,s)>0$, при достаточно маленьком $\alpha$. Значит, и во всем
цилиндре $W(x,t)\geq 0$, в частности $W(x_{s+1},t_{s+1})\geq 0$,
то есть
$$
u(x_{s+1},t_{s+1})\leq M-\alpha v(x_s,t_s)<M
$$
Что и хотели доказать.\\

\begin{theorem}[о стабилизации]
Пусть $u(x,t)\in C^{2,1}(w_\infty)\cap C(\overline w_\infty)$ --
решение уравнения $Tu=0$,
$$
w_\infty=\left\{\,(x,t)\,\Bigl|\,x\in \Omega,\,0<t<\infty,
\right\}
$$
$$
S_\infty=\left\{\,(x,t)\,\Bigl|\,x\in \partial\Omega,\,0<t<\infty,
\right\}
$$
Пусть $u|_{S_\infty}=0$, тогда $u(x,t)\rightarrow0\quad
t\rightarrow \infty$ равномерно по всем $x \in \overline\Omega$
\end{theorem}


\noindent\textbf{Доказательство:}\\
Для удобства положим $0\in\Omega$. Рассмотрим функцию
$$
v(x,t)=e^{-at}\prod_{j=1}^n \cos b x_j,\quad a>0
$$
Поскольку $v_t=-av$ и $v_{x_j x_j}=-b^2 v$, то $Tv=(nb^2-a)v$.
Следовательно при $a=nb^2$ $v(x,t)$ -- решение уравнения
теплопроводности. Выберем $b$ настолько малым, что
$$
\Omega\subset \left\{\,|x_j|<\frac \pi{4b},\quad j=1,\ldots,n
\right\}
$$
Внутри этого параллелепипеда $v(x,0)>0$, поэтому (из непрерывности
$u$ и $v$) $\exists M>0:\, |u(x,0)|<v(x,0),\quad x\in\Omega$.
Кроме того, $v|_{S_\infty}>0$. Рассмотрим функции
$$
W_1(x,t)=Mv(x,t)-u(x,t)
$$
$$
W_2(x,t)=Mv(x,t)+u(x,t)
$$
Очевидно, $TW_1=TW_2=0$, при этом $M$ мы выбрали так, что
$W_1(x,0)=Mv(x,0)+u(x,0)\geq 0$ и кроме того, $W_1|_{S_\infty}>0$,
отсюда по принципу максимума $W_1(x,t)\geq 0,\quad x\in \overline
w_\infty$, то есть $u(x,t)\leq Mv(x,t)$. Аналогично применяя
принцип максимума к $W_2$, получаем $-u(x,t)\leq Mv(x,t)$. Итак,
$$
|u(x,t)| \leq Mv(x,t)\quad x\in \overline w_\infty
$$
Но $v(x,t)$ убывает к нолю равномерно по $x\in\overline\Omega$,
так что теорема доказана.\\

\subsection {Теоремы единственности}
\begin{theorem}
Первая краевая задача для оператора теплопроводности имеет

\end{theorem}


\noindent\textbf{Доказательство:}\\

$$
0=\min_{\sigma_\tau} v(x,t)\leq v(x,t)\leq \max_{\sigma_\tau}
v(x,t)=0
$$
Поэтому $v(x,t)=0$ в $\tilde{w}_\tau$ и единственность доказана.\\
