\begin{tproblem}
При каких $A\in\R$ $\exists$ решение $u(x,t)\in C^2(x\geq 0,~y\geq 0)$ задачи:
$
\left\{
\begin{aligned}
&u_{tt}=u_{xx}+A,~~x>0,~t>0\\
&u|_{t=0}=u_t|_{t=0}=0\\
&(u_x+2u_t)|_{x=0}=\sin 2t\\
\end{aligned}
\right.
$
\end{tproblem}
\begin{solution}
Замена $u=v+\frac{At^2}{2}$ приводит нас к системе: 
$
\left\{
\begin{aligned}
&v_{tt}=v_{xx},~~x>0,~t>0\\
&v|_{t=0}=v_t|_{t=0}=0\\
&(v_x+2v_t)|_{x=0}=-2At+\sin 2t\\
\end{aligned}
\right.
$\\
\pt{1} В области x>t имеем: v=f(x-t)+g(x+t). Учитывая $v|_{t=0}=v_t|_{t=0}=0$ имеем f=-c, g=c, u=f+g=0.\\
\pt{2} В области x<t g определена, по-этому остаётся найти только f. u=f(x-t)+c. Воспользовавшись  $(v_x{+}2v_t)|_{x=0}\bw=-2At+\sin 2t$ находим $v=-\frac{\cos(2(x-t))}{2} -A(x-t)^2+C+C'$. Для непрерывности вдоль характеристики x=t положим $C+C'=\frac{1}{2}$. Вспоминая что $u=v+\frac{At^2}{2}$ имеем:\\
$$
u=
\left\{
\begin{aligned}
&\frac{At^2}{2},~x\geq t\\
&-\frac{\cos(2(x-t))}{2}-A(x-t)^2+\frac{1}{2}+\frac{At^2}{2},~x\leq t\\
\end{aligned}
\right.
$$
\pt{3} Разрывы 2х производных, если они есть, обязаны лежать на характеристике. Имеем:
$$
u_{xx}=
\left\{
\begin{aligned}
&0,~x\geq t\\
&-2A+2\cos(2(x-t)),~x>t\\
\end{aligned}
\right.
~~
u_{tt}=
\left\{
\begin{aligned}
&A,~x>t\\
&-A+2\cos(2(x-t)),~x\leq t\\
\end{aligned}
\right.
$$
Таким образом получаем, что при A=1 $u(x,t)\in C^2(x\geq 0,~y\geq 0)$.
\end{solution}
\begin{tproblem} Найти все $\alpha\in\R$, что решение задачи удовлетворяет условию $u=O(e^{-300t})$ при $t\to\infty$:
$$
\left\{
\begin{aligned}
&u_t=u_{xx}+\alpha u,~~x\in (0,\pi),~t>0\\
&u|_{t=0}=\sin x\\
&u|_{x=0}=u|_{x=\pi}=0
\end{aligned}
\right.
$$
\end{tproblem}
\begin{solution} Сведём задачу к задаче Штурма-Лиувилля.\\
\pt{1} Избавимся от $\alpha u$: $u(x,t)=e^{\alpha t}v(x,t)$.Тогда для v имеем:
$
\left\{
\begin{aligned}
&v_t=u_{xx},~~x\in (0,\pi),~t>0\\
&v|_{t=0}=\sin x\\
&v|_{x=0}=v|_{x=\pi}=0
\end{aligned}
\right.
$\\
\pt{2} Решаем задачу методом Фурье: ищем решение v(x,t)=X(x)T(t). Имеем задачу Штурма-Лиувилля:\\
\begin{center}
$
\left\{
\begin{aligned}
&X_k''-\lambda_k X_k=0\\
&X_k(0)=X_k(\pi)=0\\
\end{aligned}
\right.
$ 
   , откуда $X_k=\sin kx$, $\lambda_k=-k^2$.
\end{center}
\pt{4} Задача для $T_k$: $T_k'-\lambda_k T_k=0$. Решение $T_k=C_k e^{-k^2 t}$.\\
\pt{5} Имеем $v(x,t)=\suml{k=1}{\infty}T_k(t)X_k(x)=\suml{k=1}{\infty}C_k e^{-k^2 t}\sin kx$. Используя $v|_{t=0}=\sin x$ получаем $v(x,t)=e^{-t}\sin x$.\\
\pt{6} $u(x,t)=e^{\alpha t}v(x,t)=e^{\alpha t}e^{-t}\sin x$. Тогда $u=O(e^{-300t}) \Lra \alpha\leq -299$.
\end{solution}
\begin{tproblem} Решить задачу:  
$
\left\{
\begin{aligned}
&u_{tt}=\Delta u + e^{-t}\sin{x}\sin{2x},~~0<x,y<\pi,~t>0\\
&u|_{x=0}=u|_{x=\pi}=u|_{y=0}=u|_{y=\pi}=0\\
&u|_{t=0}=u_t|_{t=0}=0\\
\end{aligned}
\right.
$
\end{tproblem}
\begin{solution} Пусть P - прямоугольник, на котором задана задача. Чтобы решить исходную задачу, решим методом Фурье такую задачу:
$
\left\{
\begin{aligned}
-\Delta u &= \lambda u\\
u|_{\pd P} &=0\\
\end{aligned}
\right.
$\\
\pt{1} Решение, как обычно, ищем в виде $u=\suml{}{}T(t)X(x)Y(y)$. Стандартным образом разделяя переменные имеем:\\
\begin{center}
$
\left\{
\begin{aligned}
&X_k''-\mu_kX_k=0\\
&X_k(0)=X_k(\pi)=0\\
\end{aligned}
\right.
~~
\left\{
\begin{aligned}
&Y_m''-\nu_mY_m=0\\
&Y_m(0)=Y_m(\pi)=0\\
\end{aligned}
\right.
$\ \ \ 
 откуда \ \ \ 
$
\left\{
\begin{aligned}
&\mu_k=-k^2\\
&X_k=\sin{kx}\\
\end{aligned}
\right.
~~
\left\{
\begin{aligned}
&\nu_m=-m^2\\
&Y_m=\sin{ky}\\
\end{aligned}
\right.
$
\end{center}
\pt{2} Имеем $\lambda_{k,m}=-k^2-m^2$, $u_{k,m}=\sin{kx}\sin{my}$. Тогда $u=\suml{k,m=1}{\infty}T_{k,m}(t)\sin{kx}\sin{my}$ и удовлетворяет нашей системе. Тогда $T_{k,m}=0$ при $(k,m)\ne(1,2)$, то есть $u=T_{1,2}(t)\sin{x}\sin{2y}$. Подставляя в исходную систему, получаем уравнение: $T_{1,2}''+5T_{1,2}=e^{-t}$.
Решая, получаем $T_{1,2}=C_1\cos(\sqrt{5}t)+C_2\sin(\sqrt{5}t)+\frac{e^{-t}}{6}$. Используя $u|_{t=0}=u_t|_{t=0}=0$ получаем $C_1=-\frac{1}{6}$, $C_2=\frac{1}{5\sqrt{5}}$.
Итого $u=(-\frac{1}{6}\cos(\sqrt{5}t)+\frac{1}{5\sqrt{5}}\sin(\sqrt{5}t)+\frac{e^{-t}}{6})\sin{x}\sin{2y}$.
\end{solution}
\begin{tproblem} Найти решение краевой задачи:
$
\left\{
\begin{aligned}
&r^2u_{rr}+ru_r+u_{\ph\ph}=-4r^3\cos 6\ph,~~1<r<3,~0<\ph<\frac{\pi}{3}\\
&u_\ph|_{\ph=0}=u_\ph|_{\ph=\frac{\pi}{3}}=0\\
&u|_{r=3}=\cos 3\ph,~ u|_{r=1}=1\\
\end{aligned}
\right.
$
\end{tproblem}
\begin{solution} Избавимся от неоднородности в уравнении: $u=v+Cr^3\cos 6\ph$. Подставляя, находим $C=\frac{4}{27}$.\\
\pt{1} После замены $u=v+\frac{4}{27}r^3\cos 6\ph$ получаем задачу:
$$
\left\{
\begin{aligned}
(*)
&\left\{
\begin{aligned}
&r^2v_{rr}+rv_r+v_{\ph\ph}=0\\
&v_\ph|_{\ph=0}=v_\ph|_{\ph=\frac{\pi}{3}}=0\\
\end{aligned}
\right.\\
(**)&\left\{
\begin{aligned}
&v|_{r=3}=\cos{3\ph} -4\cos{6\ph}\\
&v|_{r=1}=1-\frac{4}{27}\cos{6\ph}\\
\end{aligned}
\right.
\end{aligned}
\right.
$$
\pt{2} из $(*)$:
$
\left\{
\begin{aligned}
&\Ph_k''+\lambda\Ph=0\\
&\Ph'(0)=\Ph'(\frac{\pi}{3})=0\\
\end{aligned}
\right.
$\ \ \ 
Решение здесь константа для $\lambda_0=0$ и $\Ph_k(\ph)=\cos{3k\ph}$, $\lambda_k=(3k)^2$.\\
\pt{3} Решая задачу Штурма-Лиувилля для R(r) ($r^2R_k''+rR_k'-\lambda_kR_k=0$), получаем, что $v(r,\ph)=A_0+B_0\ln r \bw+\suml{k=1}{\infty}(A_kr^{3k}+B_kr^{-3k})\cos{3k\ph}$.\\
\pt{4} Из $(**)$ получаем:
$$
\left\{
\begin{aligned}
&A_0+B_0\ln{3}=0\\
&A_0=1\\
\end{aligned}
\right.
~~
\left\{
\begin{aligned}
&A_13^3+B_13^{-3}=1\\
&A_1+B_1=0\\
\end{aligned}
\right.
~~
\left\{
\begin{aligned}
&A_23^6+B_23^{-6}=-4\\
&A_2+B_2=-\frac{4}{27}\\
\end{aligned}
\right.
~~
\left\{
\begin{aligned}
&A_k=0,~k>2\\
&B_k=0,~k>2\\
\end{aligned}
\right.
$$
$$
\left\{
\begin{aligned}
&A_0=1\\
&B_0=-\frac{1}{\ln{3}}\\
\end{aligned}
\right.
~~
\left\{
\begin{aligned}
&A_1=\frac{1}{3^3-3^{-3}}\\
&B_1=-\frac{1}{3^3-3^{-3}}\\
\end{aligned}
\right.
~~
\left\{
\begin{aligned}
&A_2=-4\frac{3^6-3^{-3}}{3^{12}-1}\\
&B_2=-4\frac{3^{-3}-3^{-6}}{1-3^{-12}}\\
\end{aligned}
\right.
~~
\left\{
\begin{aligned}
&A_k=0,~k>2\\
&B_k=0,~k>2\\
\end{aligned}
\right.
$$
Откуда, учитывая $u=v+\frac{4}{27}r^3\cos 6\ph$ имеем $u=1-\frac{\ln{r}}{\ln{3}}+\frac{r^3-r^{-3}}{3^3-3^{-3}}\cos{3\ph}-4(\frac{3^6-3^{-3}}{3^{12}-1}r^6+\frac{3^{-3}-3^{-6}}{1-3^{-12}}r^{-6}-\frac{1}{27}r^3)\cos{6\ph}$.
\end{solution}
\begin{tproblem} Найти решение уравнения Лапласа в прямоугольнике:
$
\left\{
\begin{aligned}
&\Delta u=0\\
&u_x|_{x=0}=\sin{y}, ~~ u_x|_{x=2\pi}=\sin{3y}\\
&u|_{y=0}=\cos{2x}, ~~ u|_{y=\pi}=\cos{6x}\\
\end{aligned}
\right.
$
\end{tproblem}
\begin{solution} Разобьём систему на 2 по противоположным сторонам прямоугольника (u=v+w):\\
\pt{1}
$
\left\{
\begin{aligned}
&\Delta v=0\\
&v_x|_{x=0}=0, ~~ v_x|_{x=2\pi}=0\\
&v|_{y=0}=\cos{2x}, ~~ v|_{y=\pi}=\cos{6x}\\
\end{aligned}
\right.
$ \ \ \ \ \ 
$v_k=X_k(x)Y_k(y)$, тогда $-\frac{X_k''}{X_k}=\frac{Y_k''}{Y_k}=\lambda_k$, 
$
\left\{
\begin{aligned}
&X_k''+\lambda X_k=0\\
&X'(0)=X'(2\pi)=0\\
\end{aligned}
\right.
$
Откуда $v(x,y)=\suml{k=0}{\infty}Y_k(y)\cos{\frac{kx}{2}}$. Для $k>0$ $Y_k(y)=C_ke^{\frac{ky}{2}}+D_ke^{-\frac{ky}{2}}$. Из $v|_{y=0}=\cos{2x}$, $v|_{y=\pi}=\cos{6x}$ получаем:
$
\left\{
\begin{aligned}
&C_4=\frac{1}{1-e^{4\pi}}\\
&D_4=\frac{1}{1-e^{-4\pi}}\\
\end{aligned}
\right.
~~
\left\{
\begin{aligned}
&C_{12}=\frac{1}{e^{6\pi}-e^{-6\pi}}\\
&D_{12}=-\frac{1}{e^{6\pi}-e^{-6\pi}}\\
\end{aligned}
\right.
$\ \ \ Окончательно 
$v=(\frac{e^{2y}}{1-e^{4\pi}}+\frac{e^{-2y}}{1-e^{-4\pi}})\cos{2x}+\frac{e^{6y}-e^{-6y}}{e^{6\pi}-e^{-6\pi}}\cos{6x}$.\\
\pt{2} Аналогично для 2ой системы:
$
\left\{
\begin{aligned}
&\Delta w=0\\
&w_x|_{x=0}=\sin{y}, ~~ w_x|_{x=2\pi}=\sin{3y}\\
&w|_{y=0}=0, ~~ w|_{y=\pi}=0\\
\end{aligned}
\right.
$\\
$w=\suml{k=1}{\infty}X_k(x)\sin{ky}$. Коэффициенты находятся из краевых условий. Итого $w=(\frac{e^x}{1-e^{4\pi}}-\frac{e^{-x}}{1-e^{-4\pi}})\sin{y}+\frac{e^{3x}+e^{-3x}}{3(e^{6\pi}-e^{-6\pi})}\sin{3y}$.\\
\pt{3} $u=v+w=(\frac{e^{2y}}{1-e^{4\pi}}+\frac{e^{-2y}}{1-e^{-4\pi}})\cos{2x}+\frac{e^{6y}-e^{-6y}}{e^{6\pi}-e^{-6\pi}}\cos{6x}+(\frac{e^x}{1-e^{4\pi}}-\frac{e^{-x}}{1-e^{-4\pi}})\sin{y}+\frac{e^{3x}+e^{-3x}}{3(e^{6\pi}-e^{-6\pi})}\sin{3y}$.
\end{solution}
