\section{Математическое ожидание и дисперсия оценок.\\Совместное распределение.}

\begin{problem}
Пусть $X_1\dots X_n$\т выборка из генеральной совокупности НОР величин $\Rc(0,\ta]$.
Найти математическое ожидание и дисперсию оценок параметра $\ta$:

\eqn{T_1(x)=2, \quad \dfrac{S_n}{n}=2\ol{X}, \text{ где } S_n=\suml{i=0}{n}x_i.}
\eqn{T_2(x)=\max{X_i}=X_{(n)}.}
\end{problem}
\begin{solution}

\pt{1} $EX_i=\dfrac{1}{\ta}\intl{0}{\ta}x\,dx=\left.\dfrac{x^2}{2\ta}\right|_0^{\ta}=\dfrac{\ta}{2}$\, ,

$ET_1(X)=\frac{2}{n}E(\suml{1}{n}X_i)=\frac{2}{n}\suml{1}{n}EX_i=\frac{2}{n}n\frac{\ta}{2}=\ta$
Таким образом, оценка несмещенная.

$DX_i=\dfrac{1}{\ta}\intl{0}{\ta}(x-\dfrac{\ta}{2})^2\, dx=\dfrac{\ta^2}{12}$,

$DT_1(x)=D\left(\dfrac{2}{n}\suml{1}{n}X_i\right)=\dfrac{4}{n^2}\suml{1}{n}DX_i=\dfrac{\ta^2}{3n}$,

$DT_1(x)\rightarrow 0$, при $n\rightarrow\infty$.

Значит, $T_1$ состоятельная оценка.

\pt{2} Найдем распределение $X_{(n)}$

$F(x)=P\left(X_{(n)}\le X\right)=P\left(X_1\le x,\dots X_n\le x\right)=
P\left(X_1\le x\right)\dots P\left(X_n\le x\right)= \\
=\left(P\left(X_1\le x\right)\right)^n=\dfrac{x^n}{\ta^n}$,
т.к. $P\left(X\le x\right)=\dfrac{x}{\ta}$

Значит, $p_{X_{(n)}}(x)=\begin{cases}\dfrac{nx^{n-1}}{\ta^n}, \, x\in [0, \ta) \\
0, \mbox{иначе}\end{cases}$

$EX_{(n)}=\dfrac{n}{\ta^2}\intl{0}{\ta}x^n=\dfrac{n}{n+1}\ta= \ta - \dfrac{\ta}{n+1}$

Таким образом, оценка $T_2=X{(n)}$ асимптотически несмещенная.

$DX{(n)}=E^2X_{(n)} - EX_{(n)}^2$ Найдем $E^2X_{(n)}$

$E^2X_{(n)}=\dfrac{n}{\ta^n}\intl{0}{\ta}x^{n+1}\, dx=\dfrac{n}{n+2}\ta^2$

Тогда $DX{(n)}=\dfrac{n}{n+2}\ta^2 - \dfrac{n^2}{(n+1)^2}\ta^2=\dots=\dfrac{n}{(n+1)(n+2)}\ta^2\rightarrow 0$
при $n\rightarrow\infty$

Значит, $T_2=X{(n)}$ состоятельная оценка.
\end{solution}

\begin{problem}
$X_1\dots X_{n}$ выборка из генеральной совокупности НОР величин $\Rc[\ta,1)$.
Найти математическое ожидание и дисперсию оценок параметра $\ta$:

\centerline{1)$X_{(1)}=\min{X_i}$}

\centerline{2) $\widehat{\ta}_n(x)=2\ol{X} - 1$}
\end{problem}
\begin{solution}
\pt{1}
Найдем распределение $X_{(1)}$.

$P\left(X_{(1)}\le x\right)=1 - P\left(X_{(1)}> x\right)=1- \prod\limits_1^n P\left(X_i> x\right)=
1- \prod\limits_1^n \left(1-P\left(X_i\le x\right)\right)= \\
=1- \prod\limits_1^n\left(1- \dfrac{x-\ta}{1-\ta}\right)=
1- \prod\limits_1^n\left(\dfrac{1-x}{1-\ta}\right)=1- \left(\dfrac{1-x}{1-\ta}\right)^n $

Значит, $p_{X_{(1)}}(x)=\begin{cases}\dfrac{n(1-x)^{n-1}}{(1-\ta)^n}, \, x\in [\ta, 1) \\
0, \mbox{иначе}\end{cases}$

$EX_{(1)}=\dfrac{n}{(1-\ta)^n}\intl{\ta}{1}x(1-x)^{n-1}\, dx=
\dfrac{n}{(1-\ta)^n}\left(\left.\dfrac{-x(1-x)^n}{n}\right|_{\ta}^1 + \dfrac{1}{n}\intl{\ta}{1}(1-x)^n\, dx\right)=
\ta + \dfrac{1-\ta}{n+1}=EX_{(1)}$

$EX{(1)}^2=\mbox{(путем сложных вычислений как то, интегрирования по частям, дважды.)}\\ =
\ta^2 + \dfrac{2\ta(1-\ta)}{n+1} + \dfrac{2(1-\ta)^2}{(n+1)(n+2)}$

Таким образом,
$DX{(1)}=EX{(1)}^2 - E^2X{(1)}=\dots=\dfrac{-(1-\ta)^2}{(n+1)^2(n+2)}\rightarrow 0\, \mbox{при}\, n\rightarrow\infty$
$\\$

2)$EX_i=\dfrac{1}{1-\ta}\intl{\ta}{1}x\,dx=\dfrac{1+\ta}{2}$

$EX_i^2=\dfrac{1}{1-\ta}\intl{\ta}{1}x^2\,dx=\dfrac{\ta^2 + \ta +1}{3}$

$DX_i=EX{(1)}^2 - E^2X{(1)}=\dots=\dfrac{2\ta^2-\ta-1}{6}$

Таким образом,
$E\widehat{\ta}_n=2E\ol{X} -1=\dfrac{2}{n}\suml{1}{n}EX_i -1=2EX_i -1=1+\ta -1=\ta$

$D\widehat{\ta}_n=D\dfrac{2}{n}\suml{1}{n}X_i=\dfrac{4}{n^2}\suml{1}{n}DX_i=\dfrac{4}{n}DX_i=
\dfrac{4}{n}\dfrac{2\ta^2-\ta-1}{6}=\dfrac{2}{n}\dfrac{2\ta^2-\ta-1}{3}\rightarrow 0$
при $n\rightarrow\infty$
\end{solution}

\textit{Задача 3}
$X_1$\dots$X_{n}$ выборка из генеральной совокупности НОР величин с экспоненциальным распределением
Найти математическое ожидание и дисперсию оценок параметра $\ta$


\centerline{1)$X_{(1)}=\min{X_i}$}

\centerline{2) $T_1=\ol{X} -1$}


\textit{Решение}
1)$P\left(X_1\le x\right)=\intl{\ta}{x}e^{-(t-\ta)}\, dt=\left.-e^{-(t-\ta)}\right|_{\ta}^x=
1-e^{-(x-\ta)}$

Найдем распределение $X_{(1)}$

$P\left(X_{(1)}\le x\right)=1 - P\left(X_{(1)}> x\right)=1- \prod\limits_1^n P\left(X_i> x\right)=
1- \prod\limits_1^n \left(1-P\left(X_i\le x\right)\right)= 1-\left(1-\left(1-e^{-(x-\ta)}\right)\right)^n\\=
1-e^{-n(x-\ta)}=P\left(X_{(1)}\le x\right)$

Значит, $p_{X{(1)}}(x)=\begin{cases}ne^{-n(x-\ta)}, \, x\geq \ta \\
0, x<\ta\end{cases}$

$EX_{(1)}=\intl{\ta}{\infty}nxe^{-n(x-\ta)}\,dx=\mbox{(путем сложных вычислений)}=\ta +\dfrac{1}{n}$

$EX_{(1)}^2=\intl{\ta}{\infty}nx^2e^{-n(x-\ta)}\, dx\mbox{(путем сложных вычислений)}
=\ta^2+\dfrac{2}{n}\ta+\dfrac{2}{n^2}$

$DX_{(1)}=EX_{(1)}^2 - E^2X_{(1)}=\dfrac{1}{n^2}\rightarrow 0$ при $n\rightarrow\infty$
$\\$

2)
$EX_i=\intl{\ta}{\infty}xe^{-x+\ta}\, dx=e^{\ta}\intl{\ta}{\infty}xe^{-x}\,dx=
-e^{\ta}\left(\left.xe^{-x}\right|_{\ta}^{\infty} - \intl{\ta}{\infty}e^{-x}\, dx\right)=
-e^{\ta}\left(\ta e^{-\ta} + \left.e^{-x}\right|_{\ta}^{\infty}\right)=\ta +1$

$EX_i^2=\intl{\ta}{\infty}x^2e^{-x+\ta}\, dx=\dots=\ta^2 + 2\ta +2$

$DX_i=EX_i^2 - E^2X_i=1$

$ET_1=E\left(\dfrac{1}{n}\suml{1}{n}X_i -1\right)=\dfrac{1}{n}\suml{1}{n}EX_i -1=EX_i -1=\ta$

$DT_1=D\left(\dfrac{1}{n}\suml{1}{n}X_i -1\right)=\dfrac{1}{n^2}\suml{1}{n}DX_i=\dfrac{1}{n}DX_i=\dfrac{1}{n}$
$\\$

\textit{Задача 4}
$\xi_1$, $\xi_2$ независимы.
$\xi_1\sim \Ga(p_1, 1)$,
$\xi_2\sim \Ga(p_2, 1)$.
Найти $p_{\xi_1 + \xi_2}$ - плотность распределения случайной величины
$\xi_1 + \xi_2$
$\\$

\textit{Решение}

$p_{\xi_1 +\xi_2}(x)=\intl{0}{x}p_{\xi_1}(t)p_{\xi_2}(x-t)\, dt
=\dfrac{1}{\Ga(p_1)\Ga(p_2)}\intl{0}{x}t^{p_1 -1}e^{-t}(x-t)^{p_2-1}e^{-x+t}\, dt= \\
=\dfrac{e^{-x}}{\Ga(p_1)\Ga(p_2)}\intl{0}{x}t^{p_1 -1}(x-t)^{p_2-1}\, dt
=(t=xs)=\dfrac{e^{-x}}{\Ga(p_1)\Ga(p_2)}\intl{0}{1}x^{p_1 + p_2 -1}s^{p_1 -1}(1-s)^{p_2 -1}\, ds=\\
=\dfrac{e^{-x}x^{p_1 + p_2 -1}}{\Ga(p_1)\Ga(p_2)}B(p_1, p_2)
=\dfrac{e^{-x}x^{p_1 + p_2 -1}}{\Ga(p_1+p_2)}$
$\\$


Таким образом, $\xi_1 + \xi_2\sim\Ga(p_1 +p_2, 1)$
$\\$

\textit{Задача 5}
$\xi_1$, $\xi_2$ независимы.
$\xi_1\sim \Ga(p_1, a)$,
$\xi_2\sim \Ga(p_2, a)$.
Найти $p_{\xi_1 + \xi_2}$ - плотность распределения случайной величины
$\xi_1 + \xi_2$
$\\$

\textit{Решение}
Имеем
\mln{p_{\xi_1 +\xi_2}(x)=\intl{0}{x}p_{\xi_1}(x-y)p_{\xi_2}(y)\,dy =\dfrac{1}{\Ga(p_1)\Ga(p_2)a^{p_1 +p_2}}
\intl{0}{x}(x-y)^{p_1 -1}e^{-\tfrac{(x-y)}{a}}y^{p_2-1}e^{-\tfrac{y}{a}}\,dy=\\
=\dfrac{e^{-\tfrac{x}{a}}}{\Ga(p_1)\Ga(p_2)a^{p_1 +p_2}}\intl{0}{x}(x-y)^{p_1 -1}y^{p_2-1}\, dy
=\quad \hc{y=tx}\quad =\\
=\dfrac{e^{-\tfrac{x}{a}}}{\Ga(p_1)\Ga(p_2)a^{p_1 +p_2}}(x)^{p_1 + p_2 -1}
\intl{0}{1}(1-t)^{p_1 -1}t^{p_2 -1}\, dt
=\dfrac{e^{-\tfrac{x}{a}}x^{p_1 + p_2 -1}}{\Ga(p_1)\Ga(p_2)a^{p_1 +p_2}}B(p_1, p_2)
=\dfrac{e^{-\tfrac{x}{a}}x^{p_1 + p_2 -1}}{\Ga(p_1+p_2)a^{p_1 +p_2}}.}

Таким образом, $\xi_1 + \xi_2\sim\Ga(p_1 +p_2, a)$

p(y,z) совместная плотность случайных величин $\eta$ и $\zeta$ .
Пусть $\eta$ и $\zeta$ независимы с плотностями $p_1$, $p_2$ соответственно.
Тогда $p(y,z)=p_1(y) p_2(z)$

Найти плотность распределения случайной величины
$\xi=\dfrac{\eta}{\zeta}$.
$\\$

\textit{Решение}

\mln{F_{\xi}(x)=\intl{0}{\infty}\intl{-\infty}{zx}p(y,z)\,dy\,dz + \intl{-\infty}{0}\intl{zx}{\infty}p(y,z)\,dy\,dz=\\
=\intl{0}{\infty}\intl{-\infty}{zx}p_1(y)p_2(z)\, dy\,dz + \intl{-\infty}{0}\intl{zx}{\infty}p_1(y)p_2(z)\, dy\,dz
=\intl{0}{\infty}F_1(zx)p_2(z)\,dz + \intl{-\infty}{0}\left(1-F_1(zx)\right)p_2(z)\, dz=F_{\xi}}

Тогда
\eqn{p_{\xi}=\intl{0}{\infty}zp_1(zx)p_2(z)\,dz - \intl{-\infty}{0}zp_1(zx)p_2(z)\, dz= 2\intl{0}{\infty}zp_1(zx)p_2(z)\, dz=p_{\xi}.}

\section{Несмещенность и состоятельность оценок}


\textit{Задача 12}
$X_1\dots X_{n}$ НОР, $X_i\sim N(\mu, \ta^2)$
Доказать, что
$T_n=\sqrt {\dfrac {\pi}{2}}\dfrac {1}{n}\suml{1}{n}{|X_i - \mu|}$
является несмещенной и состоятельнй (в средне квадратичкском смысле, т.е. D$T_n\rightarrow 0$) оценкой
параметра $\ta$ .

\textit{Задача 13}
$X_1\dots X_{n}$ НОР, $X_i\sim \Ga(\lambda, \ta)$
$T=\suml {1}{n}X_i$
Доказать, что
$\tau_{\al}^{*}=\dfrac {\Ga (\lambda n)}{\Ga (\lambda n -\al)}$
$T^{-\al}$
является несмещенной оценкой функции
$\tau(\ta)=\ta^{-\al}$
при $\al\le \lambda_n$
$\al\geq 0$
$\\$

\textit{Решение}
$X_i\sim \Ga(\lambda, \ta)$, тогда $T=\suml{1}{n}X_i\sim \Ga(n\lambda, \ta)$, т.е.

$p_T(x)=\begin{cases}\dfrac{e^{-\tfrac{x}{a}}x^{\lambda n -1}}{\Ga(\lambda n)\ta^{\lambda n}}, \, x>0 \\
0,\, x\le 0\end{cases}$

Надо вычислить $ET^{\al}$

$ET^{-\al}=\dfrac{1}{\Ga(\lambda n)\ta^{\lambda n}}
\intl{0}{\infty}x^{-\al}e^{-\tfrac{x}{\ta}}x^{\lambda n -1} \, dx=(u=\dfrac{x}{\ta})=
\dfrac{\ta^{-\al + \lambda n}}{\Ga(\lambda n)\ta^{\lambda n}}
\intl{0}{\infty}u^{\lambda n -\al- 1}e^u \, du=\dfrac{\ta^{-\al}\Ga(\lambda n -\al)}{\Ga(\lambda n)}$.

$E\tau_{\al}^{*}=E\left(\dfrac{\Ga(\lambda n)}{\Ga(\lambda n- \lambda)}T^{-\al}\right)=\ta^{-\al}$.

Т.о. $\tau_{\al}^{*}$ является несмещенной оценкой функции
$\tau(\ta)=\ta^{-\al}$.
$\\$


\textit{Задача 14}
$X_1\dots X_{n}$ НОР, $X_i\sim N(\mu, \ta^2)$
$T^2=\suml{1}{n}(X_i - \mu)^2$
Доказать, что несмещенной оценкой для функции
$\tau _k(\ta)= \ta ^k$
является оценка
$\tau_{k}^{*}=\dfrac {\Ga(\frac{n}{2})}{2^{\tfrac{k}{2}}\Ga(\frac {n+k}{2})}$
$T^k$

\textit{Решение}
\eqn{\dfrac{T^2}{\ta^2}=\suml{1}{n}\left(\dfrac{X_i- \mu}{\ta}\right)^2\sim\chi_n^2.}

\eqn{ET^k=\dfrac{\ta^k}{2^{\tfrac{k}{2}}\Ga(\frac{n}{2})}
\intl{0}{\infty}x^{\tfrac{k}{2} +\tfrac{n}{2} -1}e^{-\tfrac{x}{2}}\, dx=(t=\dfrac{x}{2})=
\dfrac{2^{\tfrac{k+n}{2}}\ta^k}{2^{\tfrac{k}{2}}\Ga(\frac{n}{2})}
\intl{0}{\infty}t^{\tfrac{k+n}{2} -1}e^{-t}\, dt=
\dfrac{2^{\tfrac{k}{2}\ta^k\Ga(\frac{k+n}{2})}}{\Ga(\frac{n}{2})}.}

\eqn{E\tau_{k}^{*}=E\left(\dfrac{\Ga(\frac{n}{2})}{2^{\tfrac{k}{2}}\Ga(\frac{n+k}{2})}T^k\right)
=\ta^k.}

Т.о. $\tau_k^{*}$ является несмещенной оценкой функции
$\tau_k(\ta)=\ta^k$.
$\\$

\textit{Задача 15}
Доказать, что для логистического распределения несмещенной и состоятельнй оценкой параметра $\ta$ \
является $\ol{X}$

\textit{Решение}
$f(u)=\dfrac{e^{-u}}{(1-e^{-u})^2}$ - четная функция. Тогда функция $uf(u)$ - нечетная.

\eqn{\intl{-\infty}{\infty}\dfrac{ue^{-u}}{(1- e^{-u})^2}\, du=0=

\section{Информация Фишера}

\textit{Определение}
Информация Фишера:
$$i(\ta)=E_{\ta}\left(\dfrac {\partial \ln p(x_i,\ta)}{\partial \ta}\right)^2
=-E_{\ta}\left(\dfrac{\partial^{2}\ln p(x_i,\ta)}{\partial \ta^2}\right)$$
где $p(x_i, \ta)$ плотность распределения.


\textit{Задача 20}
Вычислить информацию Фишера в модели $N(\ta, \sigma^2)$.

\textit{Решение}
$p(x, \ta)=\dfrac{1}{\sqrt{2\pi}\sigma}e^{-\tfrac{(x-\ta)^2}{2\sigma^2}}$,

$\dfrac{\partial \ln {p(x,\ta)}}{\partial\ta}=\dfrac{x-\ta}{\sigma^2}$,
$\dfrac{\partial^2 \ln p(x,\ta)}{\partial\ta^2}=-\dfrac{1}{\sigma^2} $

$i(\ta)=\dfrac{1}{\sigma^2}$

\textit{Задача 21}
Вычислить информацию Фишера в модели $N(\mu, \ta^2)$.

\textit{Решение}
$p(x, \ta)=\dfrac{1}{\sqrt{2\pi}\ta}e^{-\tfrac{(x-\mu)^2}{2\ta^2}}$,

$\ln p(x, \ta)=-\dfrac{1}{2}\ln 2\pi - \ln{\ta} -\dfrac{(x-\mu)^2}{2\ta^2}$,

\eqn{\dfrac{\partial \ln {p(x,\ta)}}{\partial\ta}=-\dfrac{1}{\ta} + \dfrac{(x-\mu)^2}{\ta^3},\quad
\dfrac{\partial^2 \ln p(x,\ta)}{\partial\ta^2}=-\dfrac{1}{\ta^2}- \dfrac{3(x-\mu)^2}{\ta^4}.}

\eqn{i(\ta)=-\dfrac{1}{\ta^2}- \dfrac{3}{\ta^4}\cdot\underbrace{\dfrac{1}{\sqrt{2\pi}\ta}
\intl{-\infty}{\infty}(x-\mu)^2e^{-\tfrac{(x-\mu)^2}{2\ta^2}}\, dx}_{\mbox{Дисперсия}=\ta^2}
=\dfrac{1}{\ta^2}+ \dfrac{3}{\ta^2}=\dfrac{2}{\ta^2}.}

\textit{Задача 22}
Вычислить информацию Фишера в модели $\Ga(\lambda ,\ta)$ для параметра $\ta$.

\textit{Решение}
$p(x)=\begin{cases} \dfrac{x^{\lambda-1}e^{-\tfrac{x}{\ta}}}{\ta^{\lambda} \Ga(\lambda)} , x> 0 \\
 0, x\le\ta\end{cases}$

$\ln p(x, \ta)=\ln\dfrac{x^{\lambda -1}e^{-\tfrac{x}{\ta}}}{\ta^{\lambda}\Ga(\lambda)}=
-\dfrac{x}{\ta} +\ln\dfrac{x^{\lambda-1}}{\Ga(\lambda)}-\lambda\ln\ta=
(\lambda -1)\ln x -\dfrac{x}{\ta} -\lambda\ln\ta -\ln\Ga(\lambda)$

$\dfrac{\partial \ln {p(x,\ta)}}{\partial\ta}=\dfrac{x}{\ta^2}-\dfrac{\lambda}{\ta}$,

$\dfrac{\partial^2 \ln p(x,\ta)}{\partial\ta^2}=-\dfrac{2x}{\ta^3}+\dfrac{\lambda}{\ta^2}$

\eqn{i(\ta)=-\dfrac{\lambda}{\ta^2} +\dfrac{2}{\ta^3}
\intl{0}{\infty}\dfrac{x^{\lambda}e^{-\tfrac{x}{\ta}}}{\ta^{\lambda}\Ga(\lambda)}\,dx=
-\dfrac{\lambda}{\ta^2} +\dfrac{2}{\ta^2\Ga(\lambda)}\intl{0}{\infty}t^{\lambda}e^{-t}\, dt=
-\dfrac{\lambda}{\ta^2} +\dfrac{2}{\ta^2}\dfrac{\Ga(\lambda +1)}{\Ga(\lambda)}=\dfrac{\lambda}{\ta^2}
=i(\ta).}

\textit{Задача 29}

Построить центральный $\ga$-доверительный интервал для $\tau=\ta^{-2}$ по выборке
$X_{1}\dots X_{n}$ в модели $N(\mu, \ta^2)$.

\textit {Решение}

\section{Проверка гипотез}
\textit{Определение}
$\hc{\ka}$\т множество критических точек, т.е. множество, где не верна основная гипотеза $H_0$.

\textit{Задача 51}
Рассмотрим модель $N(\ta_1, \ta_2^2)$

1) Проверить гипотезы \begin{center} $H_0: \ta_1=\ta_{10}$\\
$H_1: \ta_1\neq \ta_{01}$
\end{center}
2) Проверить гипотезы \begin{center} $H_0: \ta_2=\ta_{20}$\\
$H_1: \ta_2\neq \ta_{02}$
\end{center}

\textit {Решение}

1) Рассмотрим $\dfrac{\ol{X}-\ta_{10}}{\sqrt{\S_n^2\diagup (n-1)}}\sim t_{n-1}$

$P\left(-t_{\ga}\le \dfrac{\ol{X}-\ta_{10}}{\sqrt{\S_n^2\diagup (n-1)}}\le t_{\ga}\right)=\ga$

Значит $\ka=\left\{ x: \ol{X}> \ta_{10}+t_{\ga}\sqrt{(...)} ; \,
\ol{x}<\ta_{10}-t_{\ga}\sqrt{(...)}\right\} $

2) Рассмотрим $S_n^2=\dfrac{1}{n}\suml{1}{n}(X_i-\ol{X})^2$

$\dfrac{nS_n^2}{\ta^2_2}\sim\chi^2_{n-1}$

$P\left(\chi_{\tfrac{1-\ga}{2}}\le\dfrac{nS_n^2}{\ta^2_{20}}\le\chi_{\tfrac{1+\ga}{2}}\right)=\ga \Rightarrow$

$\ka=\left\{ x: S_n^2>\dfrac{1}{n}\ta_{20}^2\chi_{\tfrac{1+\ga}{2}};
 \, S_n^2<\dfrac{1}{n}\ta_{20}^2\chi_{\tfrac{1-\ga}{2}}\right\}$.

\textit{Задача 52}
$X=X_{1}\dots X_{n}$
и $Y=Y_{1}\dots Y_{n}$ две независимые выборки
$X_i\sim N(\ta_1, \sigma_1^2)$ и
$Y_i\sim N(\ta_2, \sigma_2^2)$.
$\ga$ заданно.
Проверить гипотезы \begin{center} $H_0: \tau=\ta_1-\ta_{2}=0$\\
$H_1: \tau=\ta_1- \ta_{2}\neq 0$\end{center}

\textit {Решение}

Возьмем $\xi=\dfrac{\ol{X}-\ol{Y}-\tau}{\sqrt{\dfrac{\sigma_1^2}{n}+\dfrac{\sigma_2^2}{m}}}\sim N(0,1)$

$P\left(-c_{\ga}\le\dfrac{\ol{X}-\ol{Y}-\tau}{\sqrt{\dfrac{\sigma_1^2}{n}+\dfrac{\sigma_2^2}{m}}}
\le c_{\ga}\right)=\ga$

$\tau=0,\, \ka=\left\{x,y : \ol{X}-\ol{Y}>c_{\ga}\sqrt{(...)};
 \, \ol{X}-\ol{Y}<-c_{\ga}\sqrt{(...)}\right\}$

\textit{Задача 53}
$X=X_{1}\dots X_{n}$
и $Y=Y_{1}\dots Y_{n}$ две независимые выборки.
$X_i\sim N(\ta^{(1)}, \ta_2^2)$ и
$Y_i\sim N(\ta^{(2)}, \ta_2^2)$.
$\ga$ заданно.
Проверить гипотезы \begin{center} $H_0: \tau=\ta^{(1)}-\ta^{(2)}=0$\\
$H_1: \tau=\ta^{(1)}-\ta^{(2)}\neq 0$\end{center}

\textit {Решение}

Рассмотрим $\xi=\dfrac{\ol{X}-\ol{Y}-\tau}{\ta_2}, \, \xi\sim N\left(0, \dfrac{1}{n}+\dfrac{1}{m}\right)$

$\dfrac{nS_x^2}{\ta_2^2}\sim\chi_{n-1}^2, \, \dfrac{mS_y^2}{\ta_2^2}\sim\chi_{m-1}^2$

Тогда $\dfrac{nS_x^2+mS_y^2}{\ta^2_2}\sim\chi^2_{m+n-2}$

$\dfrac{\ta_2(\ol{X}-\ol{Y}-\tau)}{\ta_2\sqrt{\dfrac{m+n}{mn}\dfrac{nS_x^2+mS_y^2}{m+n-2}}}=
\dfrac{\ol{X}-\ol{Y}-\tau}{nS_x^2+mS_y^2}\sqrt{\dfrac{(m+n-2)mn}{(m+n)}}\sim t$

$P\left(-t_{\ga}\le\dfrac{\ol{X}-\ol{Y}-\tau}{nS_x^2+mS_y^2}\sqrt{\dfrac{(m+n-2)mn}{(m+n)}}
\le t_{\ga}\right)=\ga$

$\tau=0, \, \ka=\left\{ x,y : \ol{X}-\ol{Y}>t_{\ga}\sqrt{(...)}\sqrt{(...)}; \,
\ol{X}-\ol{Y}<-t_{\ga}\sqrt{(...)}\sqrt{(...)}\right\}$
