\section{lection 5}

Напомним определение условного математического ожидания и его основные свойства.

Пусть $(\gO,\vF,P)$ --- некоторое вероятностное пространство, $\vA$ ---
$\gs$-алгебра ($\vA\subset\vF$) и $X$ --- случайная величина с $E|X|<\infty$.

\begin{Def} {\em Условное математическое ожидание} $E(X|\vA)$ является
$\vA$-измеримой функцией $\go$, задаваемой с точностью до эквивалентности
следующим соотношением:
\[
    \int_B E(X|\vA)\,dP=\int_B X\,dP,\quad\forall B\in\vA.
\]
(Существование у.м.о. вытекает из теоремы Радона-Никодима.)
\end{Def}
\begin{Def} Сужение $E(X|\vA)$ на класс индикаторов $\ind_A$, $A\in\vF$,
называется {\em условной вероятностью} события $A$ при заданной $\gs$-алгебре
$\vA$ и обозначается $P(A|\vA)$. Очевидно, что $P(A|\vA)$ --- это
$\vA$-измеримая функция, удовлетворяющая условию
\[
    \int_B P(A|\vA)\,dP=P(AB),\text{ для любого } B\in\vA.
\]
\end{Def}
{\bf Свойства условного математического ожидания.}

\begin{enumerate}
\item $E(E(X|\vA))=EX$.
\item Если $X$ является $\vA$-измеримой, то
\[
    E(X|\vA)=X\text{ п.н.}
\]
\item Если $X=C$ п.н., то $E(X|\vA)=C$ п.н.,\newline а если $X\ge Y$ п.н., то
$E(X|\vA)\ge E(Y|\vA)$ п.н.
\item Линейность у.м.о.:
\[
    E(c_1X_1+c_2X_2|\vA)=c_1E(X_1|\vA)+c_2E(X_2|\vA)\text{ п.н.}
\]
\item Если с.в. $X$ измерима относительно $\vA$, то
\[
    E(XY|\vA)=XE(Y|\vA)\text{ п.н.}
\]
\item Пусть $\gs$-алгебры $\vA_i\subset\vF$, $i=1,2$, и $\vA_1\subset\vA_2$,
тогда
\[
    E(E(X|\vA_2)|\vA_1)=E(X|\vA_1)=E(E(X|\vA_1)|\vA_2)\text{ п.н.}
\]
Если $z\colon\gO\to\gX$ некоторое отображение из $\gO$ в $\gX$, то по
определению
\[
    E(X|z)=E(X|\vF_z),
\]
где $\vF_z=\{z^{-1}(B),\ B\in\vF_z'\}$, а
$\vF'_z=\{B\subset\gX,\ z^{-1}(B)\in\vF\}$.
\item Если с.в. $X$ не зависит от $\gs$-алгебры $\vA$, то
\[
    E(X|\vA)=EX.
\]
\item Если $z$ --- случайная величина, то берется
\[
    \vF_z=\{z^{-1}(B),\ B\in\vJ^1\}\ (\vJ^1\text{ --- $\gs$-алгебра}),
\]
при этом $E(X|z)=g(z)$, где $g(\cdot)$ --- борелевская функция.
\item Справедливы также теоремы о монотонной сходимости и аналоги теорем о
сходимости Фату-Лебега:

{\bf a)} Если $0\le X_n\uparrow X$ п.н., то $0\le E(X_n|\vA)\uparrow E(X|\vA)$
п.н. В частности,
\[
    P\left(\sum_{k=1}^\infty A_k|\vA\right)=
    \sum_{k=1}^\infty P(A_k|\vA)\text{ п.н.}
\]
(напомним, что запись $\sum_k A_k$ означает, что берется объединение
$\cup_k A_k$ несовместных событий, т.е. $A_iA_j=\emptyset$, $i\ne j$).

{\bf b)} Пусть $Y$ и $Z$ интегрируемы (т.е. существуют $EY$ и $EZ$).
Если $Y\le X_n$ п.н. (или $X_n\le Z$ п.н.), то
\begin{gather*}
        E\left(\underset{n\to\infty}{\varliminf}X_n|\vA\right)\le
        \underset{n\to\infty}{\varliminf}E(X_n|\vA)
    \text{ п.н.}\\
        (\text{соотв. }\underset{n\to\infty}{\varlimsup}E(X_n|\vA)\le
        E\left(\underset{n\to\infty}{\varlimsup}X_n|\vA\right)\text{ п.н.})
\end{gather*}

В частности, если $Y\le X_n\uparrow X$ п.н. (или $Y\le X_n\le Z$ п.н и
$X_n\to X$ п.н.), то
\[
    E(X_n|\vA)\to E(X|\vA)\text{ п.н. при }n\to\infty.
\]
(Далее п.н. будет часто опускаться).
\end{enumerate}

\begin{Def} Условная вероятность $P(\cdot|\vA)$ называется {\em регулярной},
если при каждом $\go$, за исключением множества меры 0, она является
вероятностной мерой.
\end{Def}

Таким образом, {\em регулярная условная вероятность} $P^\vA$ со значениями
$P(A|\vA)(\go)$ это функция, определенная на $\vF\times\gO$, обладающая
следующими свойствами:

{\bf 1)} $P(A|\vA)(\go)$ есть $\vA$-измеримая по $\go$ функция для каждого
фиксированного $A$ и представляет собой вероятность на $\vF$ при каждом
фиксированном $\go$.

{\bf 2)} Для любых фиксированных $A\in\vF$ и $B\in\vA$
\[
    P(AB)=\int_B P(A|\vA)\,dP.
\]
\begin{enumerate}
\setcounter{enumi}{8}
\item Если $P^\vA$ --- регулярная условная вероятность, то
\[
    E(X|\vA)=\int X\,dP^\vA\text{ п.н.}
\]
для любой с.в. $X$ с $E|X|<\infty$.
\end{enumerate}

\begin{Def} {\em Потоком} называется неубывающее семейство $\gs$-алгебр
$\{\vF_t,\ t\ge 0\}$, т.е. $\vF_{t_1}\subset\vF_{t_2}$ при $t_1<t_2$,
$\vF_t\subset\vF$ для любого $t$.
\end{Def}

Предположим, что $X_t(\go)$ при любом $t$ принимают значения в измеримом
пространстве $(\gX,\vB)$.

\begin{Def}Случайный процесс $X=\{X_t,\ t\ge 0\}$ называется {\em
согласованным} с потоком $\gs$-алгебр $\{\vF_t,\ t\ge 0\}$ (или {\em
адаптированным} к потоку), если с.в. $X_t$ является $\vF_t$-измеримой при
любом $t\ge 0$.
\end{Def}

\begin{Def} Случайный процесс $(X_t,\vF_t)_{t\ge 0}$ называется {\em марковским
относительно семейства $\gs$-алгебр $\{\vF_t,\ t\ge 0\}$}, если процесс
адаптирован к потоку, и для любого $t$ $\gs$-алгебры $\vF_t$ и $\vF_{\le t}$
условно независимы при данной с.в. $X_t$, т.е.
\begin{enumerate}
\item $X_t$ --- $\vF_t$-измерима при любом $t\ge 0$.
\item $P(AB|X_t)=P(A|X_t)P(B|X_t)$ $\forall A\in\vF_{\ge t},B\in\vF_t,t\ge 0$.
\end{enumerate}
\end{Def}

\begin{problem} Проверить, что случайный процесс $X$ марковский относительно
семейства $(\vF_t,\ t\ge 0)$ является просто марковским (т.е. относительно
семейства $\vF_{\le t}$).
\end{problem}

\begin{Lemma} Пусть процесс $\{X_t,\ t\ge 0\}$ адаптирован к потоку
$\{\vF_t,\ t\ge 0\}$. Тогда следующие условия эквивалентны:
\begin{enumerate}
\item Случайный процесс $\{X_t,\ t\ge 0\}$ марковский относительно семейства
$\gs$-алгебр $\{\vF_t,\ t\ge 0\}$.
\item Для любого $t\ge 0$ и произвольной $\vF_{\ge t}$-измеримой ограниченной
случайной величины $Y$ выполнено равенство
\[
    E(Y|\vF_t)=E(Y|X_t)\text{ (п.н.)}.
\]
\item Для любой измеримой ограниченной функции $f(x)$ ($\sup_x|f(x)|<\infty$) и
произвольных $s\ge t$ верно
\[
    E(f(X_s)|\vF_t)=E(f(X_s)|X_t)\text{ (п.н.)}.
\]
\end{enumerate}
\end{Lemma}

\begin{proof} Установим $1\implies 2$. Так как любая ограниченная
$\vF_{\ge t}$-измеримая с.в. может быть представлена как предел простых
функций, т.е. конечных линейных комбинаций индикаторов, то достаточно проверить
требуемое свойство для $Y=\ind_A$, где $A\in\vF_{\ge t}$, а затем
воспользоваться свойствами у.м.о.

Итак, проверим, что
\[
    P(A|\vF_t)=P(A|X_t),\ A\in\vF_{\ge t}.
\]
С одной стороны, в силу марковости (и свойств у.м.о.) имеем цепочку равенств
\begin{align*}
    P(AB)&=E(P(AB|X_t))=\\
    &=E(P(A|X_t)P(B|X_t))=E(P(A|X_t)E(\ind_B|X_t))=\\
    &=E(E(\ind_B P(A|X_t)|X_t))=E(\ind_B P(A|X_t)).\\
\intertext{С другой стороны,}
    P(AB)&=E\ind_A\ind_B=E(E(\ind_A\ind_B|\vF_t))=\\
    &=E(\ind_B E(\ind_A|\vF_t))=E(\ind_B P(A|\vF_t)).
\end{align*}
Таким образом, для любого $B\in \vF_t$
\[
    \int_B P(A|X_t)\,dP=\int_B P(A|\vF_t)\,dP\qquad(=P(AB)),
\]
а поскольку $P(A|X_t)$ --- это $\vF_t$-измеримая функция, получаем необходимое
равенство
\[
    E(Y|\vF_t)=E(Y|X_t)\text{ для }Y=\inf_A,\ A\in\vF_{\ge t}.
\]

Теперь покажем, что $2\implies 1$. Пусть $A\in\vF_{\ge t}$ и $B\in\vF_t$, тогда
\begin{align*}
    P(AB|X_t)&=E(\ind_A\ind_B|X_t)=E(E(\ind_A\ind_B|\vF_t)|X_t)=\\
    &=E(\ind_B E(\ind_A|\vF_t)|X_t)=E(\ind_B E(\ind_A|X_t)|X_t)=\\
    &=E(\ind_A|X_t)E(\ind_B|X_t)=P(A|X_t)P(B|X_t).
\end{align*}

Так как 3 --- это частный случай 2 (при $Y=f(X_1)$), то надо доказать лишь, что
$3\implies 2$.

Пусть сначала $Y=f_1(X_{s_1}\dots f_n(X_{s_n})$, где $t\le s_1<\dots<s_n$ и
$\sup_x|f_i(x)|<\infty$, $i=\overline{1,n}$. Установим интересующий нас
результат по индукции. При $n=1$ утверждение справедливо, так как совпадает с
3. Предположим, что для $n-1$ равенство установлено, и проверим его для $n$.
Имеем
\begin{align*}
        E\left(\prod_{i=1}^{n-1}f_i(X_i)g(X_{s_{n-1}})|X_t\right)
        &=E\left(\prod_{i=1}^{n-1}f_i(X_i)
        E(f_n(X_{s_n})|\vF_{s_{n-1}})|X_t\right)=\\
        &=E\left(E\left(\prod_{i=1}^n
    f_i(X_{s_i})|\vF_{s_{n-1}}\right)|X_t\right)=\\
        &=E\left(\prod_{i=1}^n f_i(X_{s_i})|X_t\right)=E(Y|X_t).
\end{align*}
Доказательство закончено, так как любую $\vF_{\ge t}$-измеримую ограниченную
с.в. можно приблизить с помощью $\prod_{i=1}^n f_i(X_{s_i})$.
\end{proof}

Для любого марковского процесса справедлив следующий результат.
\begin{Lemma} Процесс $\{X_t,\ t\ge 0\}$ марковский тогда и только тогда, когда
для любой измеримой ограниченной $f(x)$ и произвольного набора
$t_1\le t_2\le\dots\le t_n\le s$ с вероятностью 1
\[
    E(f(X_s)|X_{t_1},\dots,X_{t_n})=E(f(X_s)|X_{t_n}).
\]
\end{Lemma}
\begin{proof} Если процесс марковский, то требуемое утверждение вытекает из
предыдущей леммы. В самом деле
\begin{multline*}
        E(f(X_s)|X_{t_1},\dots,X_{t_n})=
        E(E(f(X_s)|\vF_{\le t_n})|X_{t_1},\dots,X_{t_n})=\\
        =E(E(f(X_s)|X_{t_n})|X_{t_1},\dots,X_{t_n})=
        E(f(X_s)|X_{t_n}).
\end{multline*}

Обратно, пусть указанные у.м.о. совпадают, покажем, что тогда
\[
        E(f(X_s)|\vF_{\le t})=E(f(X_s)|X_t)\text{ при }s\ge t.
\]
Для этого достаточно проверить, что для любого $B\in\vF_{\le t}$
\[
        \int_B f(X_s)\,dP=\int_B E(f(X_s)|X_t)\,dP.
\]
В силу условий леммы эти интегралы совпадают для
$B\in\gs(X_{t_1},\dots,X_{t_n},X_t)$ при $t_1\le\dots\le t_n\le t\le s$.
Правая и левая части равенства --- это конечные меры (не обязательно
вероятностные), совпадающие на цилиндрах, порождающих $\vF_{\le t}$. В силу
единственности продолжения меры равенство будет выполнено для любого
$B\in\vF_{\le t}$.
\end{proof}

Итак, пусть имеется измеримое пространство $(\gX,\vB)$, в котором все
одноточечные множества измеримы, называемое {\em фазовым}. И пусть
$(X_t,\vF_t)_{t\ge 0}$ --- марковский процесс относительно потока
$\{\vF_t,\ t\ge 0\}$ со значениями в фазовом пространстве. Тогда с
вероятностью 1 при $t\ge s$ для любого $A\in\vB$
\[
    P(X_t\in A|\vF_s)=P(X_t\in A|X_s).
\]
В силу свойства 7 у.м.о. существует такая функция $P(s,x,t,A)$, что
\[
    P(X_t\in A|X_s)=P(s,X_s,t,A).
\]
Эта функция играет важную роль в теории марковских процессов. Но для
плодотворной теории надо наложить дополнительные требования. Они станут
особенно понятными, если вспомнить следующую интерпретацию
\[
    P(s,x,t,A)=P(X_t\in A|X_s=x).
\]
\begin{Def} Функция $P(s,x,t,A)$ называется {\em марковской переходной
функцией} на $(\gX,\vB)$, если
\begin{enumerate}
\renewcommand{\labelenumi}{\theenumi $^\circ$}
\item для любых $s$, $x$, $t$ (как функция $A$) $P(s,x,t,\cdot)$ ---
вероятностная мера на $\vB$,
\item для любых $s$, $x$, $A$ (как функция $x$) $P(s,\cdot,t,A)$ измерима,
\item
\[
        P(s,x,s,A)=\gd_x(A),\text{ здесь }\gd_x(A)=\begin{cases}
        1,&\text{при }x\in A,\\ 0,&\text{при }x\notin A,\end{cases}
\]
\item выполнено уравнение Колмогорова-Чепмена, т.е. для любых
$0\le s\le u\le t$
\[
        P(s,x,t,A)=\int_\gX P(s,x,u,dy)P(u,y,t,A).
\]
\end{enumerate}
(Существует такой подход, при котором изучается это семейство функций, а
точнее, порождаемое ими семейство линейных операторов. При этом не
предполагается существование ни вероятностного пространства, ни марковского
случайного процесса.)
\end{Def}

Действительно, с измеримым пространством $(\gX,\vB)$ связаны два банаховых
пространства.

$\BB$ --- совокупность ограниченных $\vB$-измеримых функций $x\in\gX$, норма
определена следующим образом:
\[
    \|f\|=\sup_{x\in\gX}|f(x)|.
\]

$\VV$ --- совокупность обобщенных мер (или зарядов), т.е. числовая
счетно-аддитивная функция множеств $A\in\vB$, норма $\gv$ --- это полная
вариация на всем пространстве:
\[
    \|\gv\|=|\gv|(\gX).
\]

Оказывается, что между $\BB$ и $\VV$ существует определенная связь:
$\VV\subset\BB^*$ и $\BB\subset\VV^*$ (где знак $*$ показывает, что речь идет
о сопряженном пространстве).

В самом деле, положим
\[
    \left<\gv,f\right>=\int_\gX\gv(dx)f(x),
\]
где интеграл определяется следующим образом
\[
    \int_\gX\gv(dx)f(x)=\int_\gX\gv^+(dx)f(x)-\int_\gX\gv^-(dx)f(x),
\]
а $\gv=\gv^+-\gv^-$ --- это разложение Жордана.

Тогда каждому элементу $\gv\in\VV$ соответствует линейный функционал
$\left<\gv,\cdot\right>$ на $\BB$, а каждому элементу $f\in\BB$ соответствует
линейный функционал $\left<\cdot,f\right>$ на $\VV$.

\begin{problem} Доказать, что норма элемента и норма соответствующего линейного
функционала совпадают:
\[
    \|\gv\|=\sup_{\|f\|=1}|\left<\gv,f\right>|,\quad
    \|f\|=\sup_{\|\gv\|=1}|\left<\gv,f\right>|.
\]
\end{problem}

Линейные операторы в пространстве $\BB$ будем записывать слева от элемента
$f\in\BB$, а в пространстве $\VV$ --- справа.

Пусть $P(s,x,t,A)$ --- марковская переходная функция, удовлетворяющая
требованиям $1^\circ-4^\circ$. Определим на пространстве $\BB$ семейство
операторов $P^{st}$ ($s\le t$, $s,t\in T$) с помощью соотношения
\[
    P^{st}f(x)=\int_\gX P(s,x,t,dy)f(y).
\]
(Существование и ограниченность интервала для $f\in\BB$ обеспечивается тем, что
$P(s,x,t,\cdot)$ --- конечная мера (свойство $1^\circ$), а измеримость
$P^{st}f(x)$ по $x$ --- измеримостью $P(s,\cdot,t,A)$ (свойство
$2^\circ$)).

Установим свойства операторов $P^{st}$.

1) В силу их определения операторы {\em линейны}.

Остальные свойства операторов вытекают из свойств переходной функции.

2) Операторы $P^{st}$ {\em сжимающие}.

В само деле, так как $P(s,x,t,\cdot)$ --- вероятностная мера ($1^\circ$), то
\[
    |P^{st}f(x)|\le\int P(s,x,t,dy)\|f\|=\|f\|,
\]
иначе говоря
\[
    \|P^{st}f\|\le\|f\|,\text{ т.е. }\|P^{st}\|\le 1.
\]

3) Операторы {\em сохраняют положительность}, т.е. неотрицательные функции
переводят в неотрицательные. Действительно, опять-таки в силу $1^\circ$, если
$f(x)\ge 0$, то $P^{st}f(x)\ge 0$.

4) $P^{st}1\equiv 1$, это также следствие $1^\circ$.

5) $P^{ss}=E$ (тождественный оператор).

Это вытекает из $3^\circ$, так как
\[
    P^{ss}f(x)=\int\gd_x(dy)f(y)=f(x).
\]

6) $P^{st}=P^{su}P^{ut}$ при $s\le u\le t$. В самом деле, уравнение
Колмогорова-Чепмена ($4^\circ$) дает
\begin{multline*}
    P^{st}f(x)=\int P(s,x,t,dy)f(y)=
    \iint P(s,x,u,dz)P(u,z,t,dy)f(y)=\\
    =\int P(s,x,u,dz)\int P(u,z,t,dy)f(y)=
    P^{su}(P^{ut}f)(x).
\end{multline*}
(В тех случаях, когда интегрирование ведется по всему пространству $\gX$,
часто будем для простоты писать $\int$ вместо $\int_\gX$).

В пространстве $\VV$ введем операторы $P^{st}$ ($s\le t$, $s,t\in T$) с
помощью соотношения
\[
    \gv P^{st}(A)=\int\gv(dx)P(s,x,t,A).
\]
(Существование интеграла обеспечивается свойством $2^\circ$ --- измеримостью
по $x$ переходной функции, а счетная аддитивность $\gv P^{st}$ свойством
$1^\circ$, т.е. счетной аддитивностью $P(s,x,t,\cdot)$).

Свойства $1')$--$6')$ операторов $P^{st}$ в пространстве $\VV$ аналогичны
свойствам 1)--6).
\begin{itemize}
\item[$1'$)] Операторы $P^{st}$ {\em линейны} в силу определения.
\item[$2'$)] Операторы {\em сжимающие}, поскольку в силу $1^\circ$ получаем
\[
    \|\gv P^{st}\|\le\|\gv\|.
\]
\item[$3'$)] Меры переводятся в меры.
\item[$4'$)] $\gv P^{st}(\gX)=\gv(\gX)$.\newline
(Эти два свойства справедливы также в силу $1^\circ$).
\item[$5'$)] $P^{ss}=E$ следует из $3^\circ$.
\item[$6'$)] $P^{st}=P^{sk}P^{ut}$ для $s\le u\le t$.
\end{itemize}

В самом деле, уравнения Колмогорова-Чепмена превращается в соотношение
\[
    \gv P^{st}=(\gv P^{sk})P^{ut},
\]
т.е. по форме $6'$) совпадает с $6$). Однако порядок применения операторов
здесь другой (сначала $P^{su}$, а потом $P^{ut}$).

Заметим далее, что операторы $P^{st}$ в пространствах $\BB$ и $\VV$
{\em сопряжены} друг другу, поскольку для $f\in\BB$, $\gv\in\VV$
\[
    \left<\gv,P^{st}f\right>=\left<\gv P^{st},f\right>,
\]
так как правая и левая части равны
\[
    \iint\gv(dx)P(s,x,t,dy)f(y).
\]
(Более точно, оператор $P^{st}$ на $\BB$ --- это сужение оператора в $\VV^*$,
сопряженного к оператору $P^{st}$ в $\VV$, и наоборот).

Доказать, что $\|P^{st}\|=1$.
\end{problem}

функцией.
